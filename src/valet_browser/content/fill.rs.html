1//! Fill detected form fields with credentials.
2//!
3//! Uses the native input setter and dispatches `input` + `change` events
4//! so that JavaScript frameworks (React, Vue, etc.) pick up the change.
5
6use wasm_bindgen::prelude::*;
7use web_sys::HtmlInputElement;
8
9/// Set the value of an input field, dispatching synthetic events.
10pub(crate) fn set_field_value(field: &HtmlInputElement, value: &str) {
11    let field_id = field.id();
12    // Use the native HTMLInputElement.value setter to bypass
13    // React/Vue controlled-component guards.
14    if let Some(setter) = native_value_setter(field) {
15        tracing::trace!(id = %field_id, "using native setter");
16        setter.call1(field, &JsValue::from_str(value)).ok();
17    } else {
18        tracing::trace!(id = %field_id, "using fallback set_value");
19        field.set_value(value);
20    }
21
22    let _ = field.dispatch_event(
23        &web_sys::Event::new_with_event_init_dict("input", &event_init(true)).unwrap(),
24    );
25    let _ = field.dispatch_event(
26        &web_sys::Event::new_with_event_init_dict("change", &event_init(true)).unwrap(),
27    );
28    tracing::trace!(id = %field_id, "dispatched input + change events");
29}
30
31/// Look up the native `value` setter on the HTMLInputElement prototype
32/// so we can bypass framework property descriptors.
33fn native_value_setter(el: &HtmlInputElement) -> Option<js_sys::Function> {
34    let proto = js_sys::Object::get_prototype_of(el.as_ref());
35    let desc = js_sys::Object::get_own_property_descriptor(&proto, &JsValue::from_str("value"));
36    if desc.is_undefined() {
37        return None;
38    }
39    js_sys::Reflect::get(&desc, &JsValue::from_str("set"))
40        .ok()
41        .and_then(|v| v.dyn_into::<js_sys::Function>().ok())
42}
43
44fn event_init(bubbles: bool) -> web_sys::EventInit {
45    let init = web_sys::EventInit::new();
46    init.set_bubbles(bubbles);
47    init
48}
