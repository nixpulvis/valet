26#[derive(Serialize)]
27struct TabsQuery {
28    active: bool,
29    #[serde(rename = "currentWindow")]
30    current_window: bool,
31}

33/// Return the hostname of the active tab in the current window, if any.
34pub async fn current_tab_domain() -> Option<String> {
35    let filter = serde_wasm_bindgen::to_value(&TabsQuery {
36        active: true,
37        current_window: true,
38    })
39    .ok()?;
40    let promise = tabs_query(filter).ok()?;
41    let tabs = JsFuture::from(promise).await.ok()?;
42    let tab = js_sys::Array::from(&tabs).get(0);
43    if tab.is_undefined() {
44        return None;
45    }
46    let url = js_sys::Reflect::get(&tab, &JsValue::from_str("url")).ok()?;
47    let url = url.as_string()?;
48    web_sys::Url::new(&url)
49        .ok()
50        .map(|u| u.hostname().to_lowercase())
51}

53/// Check whether the extension has host permissions for all sites.
54pub async fn has_host_permissions() -> bool {
55    let perms = origins_object();
56    let Ok(promise) = permissions_contains(perms) else {
57        return false;
58    };
59    JsFuture::from(promise)
60        .await
61        .ok()
62        .and_then(|v| v.as_bool())
63        .unwrap_or(false)
64}
65
66/// Request host permissions for all sites. Must be called from a user gesture.
67pub async fn request_host_permissions() -> bool {
68    let perms = origins_object();
69    let Ok(promise) = permissions_request(perms) else {
70        return false;
71    };
72    JsFuture::from(promise)
73        .await
74        .ok()
75        .and_then(|v| v.as_bool())
76        .unwrap_or(false)
77}
78
79fn origins_object() -> JsValue {
80    let origins = js_sys::Array::new();
81    origins.push(&JsValue::from_str("*://*/*"));
82    let obj = js_sys::Object::new();
83    js_sys::Reflect::set(&obj, &JsValue::from_str("origins"), &origins).unwrap();
84    obj.into()
85}

87const DISMISSED_KEY: &str = "permissions_banner_dismissed";
88
89/// Check whether the user previously dismissed the permissions banner.
90pub async fn permissions_banner_dismissed() -> bool {
91    let Ok(promise) = storage_local_get(JsValue::from_str(DISMISSED_KEY)) else {
92        return false;
93    };
94    let Ok(result) = JsFuture::from(promise).await else {
95        return false;
96    };
97    js_sys::Reflect::get(&result, &JsValue::from_str(DISMISSED_KEY))
98        .ok()
99        .and_then(|v| v.as_bool())
100        .unwrap_or(false)
101}
102
103/// Persist that the user dismissed the permissions banner.
104pub async fn dismiss_permissions_banner() {
105    let obj = js_sys::Object::new();
106    js_sys::Reflect::set(&obj, &JsValue::from_str(DISMISSED_KEY), &JsValue::TRUE).unwrap();
107    if let Ok(promise) = storage_local_set(obj.into()) {
108        let _ = JsFuture::from(promise).await;
109    }
110}

112/// Write `text` to the system clipboard via `navigator.clipboard.writeText`.
113pub async fn copy_to_clipboard(text: &str) -> Result<(), JsValue> {
114    let clipboard = web_sys::window()
115        .ok_or_else(|| JsValue::from_str("no window"))?
116        .navigator()
117        .clipboard();
118    JsFuture::from(clipboard.write_text(text)).await.map(|_| ())
