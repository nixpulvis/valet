1//! Browser extension for the [valet] password manager.
2//!
3//! Compiled to WebAssembly and loaded by the extension's `background.js` and
4//! `popup.html`. The two WASM entry points are [`background::start_background`]
5//! and [`popup::start_popup`].
6//!
7//! Communication with the valet database goes through a native-messaging host
8//! (`com.nixpulvis.valet`), with the background script acting as a multiplexing
9//! RPC bridge between the popup UI and the host process.
10
11use valet::password::Password;
12use wasm_bindgen::prelude::*;
13
14pub mod background;
15pub mod content;
16pub mod logging;
17pub mod popup;
18pub mod rpc;
19
20/// Check whether a password candidate meets valet's minimum requirements.
21///
22/// Used by the popup to validate input before sending a `create_record` RPC.
23#[wasm_bindgen]
24pub fn password_is_valid(candidate: &str) -> bool {
25    match Password::try_from(candidate) {
26        Ok(password) => password.is_valid(),
27        Err(_) => false,
28    }
29}
30
31/// Generate a random 20-character password.
32#[wasm_bindgen]
33pub fn generate_password() -> String {
34    Password::generate().as_str().to_string()
35}
36
37/// WASM module initialization hook
38///
39/// This runs automatically when the module is instantiated and sets up the
40/// panic hook and logging system.
41#[wasm_bindgen(start)]
42pub fn main() {
43    console_error_panic_hook::set_once();
44
45    let subsystem = web_sys::window()
46        .and_then(|w| w.location().pathname().ok())
47        .and_then(|p| p.rsplit('/').next().map(str::to_owned))
48        .and_then(|f| f.split('.').next().map(str::to_owned))
49        .unwrap_or_else(|| "unknown".into());
50    logging::init(&subsystem);
51}
