37impl<'a> View for Locked<'a> {
38    fn show(&mut self, ctx: &Context) {
39        let id = Id::new("locked_view");
40        let mut state = State::load(ctx, id);
41
42        CentralPanel::default().show(ctx, |ui| {
43            if let Some(username) = self.login_inbox.read(ui).last() {
44                *self.active_user = Some(username);
45
46                // TODO: Make some kind of transition function to handle the
47                // logic between Locked and Unlocked. Unlocked should have one
48                // as well. In general with more than two view options this
49                // requires more thought.
50
51                // TODO: Do we clear the username or not?
52                ui.ctx().data_mut(|d| d.clear());
53                ui.ctx()
54                    .send_viewport_cmd(ViewportCommand::InnerSize(UNLOCKED_DEFAULT_SIZE.into()));
55                ui.ctx().send_viewport_cmd(ViewportCommand::Resizable(true));
56            }
57
58            ui.label("Username:");
59            let username_re = ui.add(TextEdit::singleline(&mut state.username));
60            ui.label("Password:");
61            // TODO: Update PasswordInput to operate on Password directly.
62            let password_re = ui.add(PasswordInput::new(&mut state.password));
63            ui.add_space(5.);
64            ui.horizontal(|ui| {
65                if ui.add(Button::new("Unlock")).clicked()
66                    || password_re.lost_focus()
67                        && username_re.ctx.input(|i| i.key_pressed(Key::Enter))
68                    || username_re.lost_focus()
69                        && password_re.ctx.input(|i| i.key_pressed(Key::Enter))
70                {
71                    // XXX: This is obviously hacky, but I don't want to deal with sharing things now.
72                    let username = state.username.clone();
73                    let password = state.password.clone();
74                    let client = self.client.clone();
75                    let tx = self.login_inbox.sender();
76                    self.rt.spawn(async move {
77                        if client.unlock(username.clone(), password).await.is_ok() {
78                            tx.send(username).ok();
79                        }
80                    });
81                }
82                if ui.add(Button::new("Create")).clicked() {
83                    // XXX: This is obviously hacky, but I don't want to deal with sharing things now.
84                    let username = state.username.clone();
85                    let password = state.password.clone();
86                    let client = self.client.clone();
87                    let tx = self.login_inbox.sender();
88                    self.rt.spawn(async move {
89                        if client.register(username.clone(), password).await.is_ok() {
90                            tx.send(username).ok();
91                        }
92                    });
93                }
94            })
95        });
96        state.store(ctx, id);
97    }
98}
99
100struct State {
101    username: String,
102    password: Password,
103}
104
105impl State {
106    fn load(ctx: &Context, id: Id) -> Self {
107        State {
108            username: ctx.data(|d| d.get_temp(id.with("username")).unwrap_or_default()),
109            password: ctx.data(|d| {
110                d.get_temp(id.with("password"))
111                    .unwrap_or(Password::default())
112            }),
113        }
114    }
115
116    fn store(self, ctx: &Context, id: Id) {
117        ctx.data_mut(|d| {
118            d.insert_temp(id.with("username"), self.username);
119            d.insert_temp(id.with("password"), self.password);
120        });
121    }
122}
