16enum PasswordEvent {
17    Copy(Password),
18    Show(Password),
19}

21pub struct RecordRow<'a> {
22    label: &'a Label,
23    record_uuid: &'a Uuid<Record>,
24    username: String,
25    client: &'a Arc<EmbeddedHandler>,

29impl<'a> RecordRow<'a> {

47impl egui::Widget for RecordRow<'_> {
48    fn ui(self, ui: &mut egui::Ui) -> egui::Response {
49        let id = ui.make_persistent_id(("record", self.record_uuid.to_string()));
50        let expanded_id = id.with("expanded");
51        let shown_pw_id = id.with("shown_pw");
52        let pw_inbox_id = id.with("pw_inbox");
53
54        let expanded = ui.data(|d| d.get_temp::<bool>(expanded_id).unwrap_or(false));
55
56        let pw_inbox: Arc<UiInbox<PasswordEvent>> =
57            ui.data_mut(|d| d.get_temp(pw_inbox_id).unwrap_or_default());
58
59        for event in pw_inbox.read(ui.ctx()) {
60            match event {
61                PasswordEvent::Copy(pw) => ui.ctx().copy_text(pw.to_string()),
62                // TODO: The revealed Password sits in egui's temp data until
63                // the row collapses or egui evicts it. Zeroizes on drop, but
64                // we should auto-evict after an idle window.
65                PasswordEvent::Show(pw) => ui.data_mut(|d| d.insert_temp(shown_pw_id, pw)),
66            }
67        }
68
69        ui.data_mut(|d| d.insert_temp(pw_inbox_id, pw_inbox.clone()));
70
71        let (primary, secondary) = match self.label.name() {
72            LabelName::Domain { id, domain } => (
73                domain.as_str().to_string(),
74                self.label
75                    .extra()
76                    .get("username")
77                    .map(String::as_str)
78                    .unwrap_or(id.as_str())
79                    .to_string(),
80            ),
81            LabelName::Simple(s) => (
82                s.clone(),
83                self.label
84                    .extra()
85                    .get("username")
86                    .cloned()
87                    .unwrap_or_default(),
88            ),
89        };
90
91        ui.vertical(|ui| {
92            ui.horizontal(|ui| {
93                let copy_width = button_width(ui, &["Copy"]);
94                let spacing = ui.spacing().item_spacing.x;
95                let label_width = (ui.available_width() - copy_width - spacing).max(0.);
96                let row_height =
97                    ui.spacing().interact_size.y + ui.text_style_height(&egui::TextStyle::Small);
98
99                // allocate_space advances the cursor by exactly label_width.
100                // new_child renders into that rect with an explicit top-down layout
101                // so the primary and secondary lines stack without touching the
102                // cursor again.
103                let (_, label_rect) = ui.allocate_space(egui::vec2(label_width, row_height));
104                let inner = ui
105                    .new_child(
106                        egui::UiBuilder::new()
107                            .max_rect(label_rect)
108                            .layout(egui::Layout::top_down(egui::Align::LEFT)),
109                    )
110                    .scope(|ui| {
111                        ui.spacing_mut().item_spacing.y = 0.;
112                        let p = ui.add(
113                            egui::Label::new(primary)
114                                .truncate()
115                                .sense(egui::Sense::click()),
116                        );
117                        let s = ui.add(
118                            egui::Label::new(
119                                egui::RichText::new(secondary)
120                                    .small()
121                                    .color(ui.visuals().weak_text_color()),
122                            )
123                            .truncate()
124                            .sense(egui::Sense::click()),
125                        );
126                        (p, s)
127                    })
128                    .inner;
129                let resp = inner.0.union(inner.1);
130                if resp.clicked() {
131                    ui.data_mut(|d| d.insert_temp(expanded_id, !expanded));
132                    if expanded {
133                        ui.data_mut(|d| d.remove::<Password>(shown_pw_id));
134                    }
135                }
136                if resp.hovered() {
137                    ui.ctx()
138                        .output_mut(|o| o.cursor_icon = egui::CursorIcon::PointingHand);
139                }
140
141                if ui
142                    .add(egui::Button::new("Copy").min_size(egui::vec2(copy_width, 0.)))
143                    .clicked()
144                {
145                    spawn_fetch(
146                        self.rt,
147                        self.client.clone(),
148                        self.username.clone(),
149                        self.record_uuid.clone(),
150                        pw_inbox.sender(),
151                        PasswordEvent::Copy,
152                    );
153                }
154            });
155
156            if expanded {
157                egui::Frame::NONE
158                    .inner_margin(egui::Margin {
159                        left: 0,
160                        right: 0,
161                        top: 2,
162                        bottom: 4,
163                    })
164                    .show(ui, |ui| {
165                        ui.horizontal(|ui| {
166                            let btn_width = button_width(ui, &["Show", "Hide"]);
167                            let spacing = ui.spacing().item_spacing.x;
168                            // With min_size on the button it renders at exactly btn_width,
169                            // so one spacing gap is the correct reservation.
170                            let text_width =
171                                (ui.available_width() - btn_width - spacing * 2.).max(0.);
172
173                            let shown_pw = ui.data(|d| d.get_temp::<Password>(shown_pw_id));
174                            let is_shown = shown_pw.is_some();
175                            let mut pw = shown_pw.unwrap_or_else(|| "xxxxxxxx".try_into().unwrap());
176                            ui.add(
177                                egui::TextEdit::singleline(&mut pw)
178                                    .password(!is_shown)
179                                    .interactive(false)
180                                    .desired_width(text_width),
181                            );
182
183                            let toggle_label = if is_shown { "Hide" } else { "Show" };
184                            if ui
185                                .add(
186                                    egui::Button::new(toggle_label)
187                                        .min_size(egui::vec2(btn_width, 0.)),
188                                )
189                                .clicked()
190                            {
191                                if is_shown {
192                                    ui.data_mut(|d| d.remove::<Password>(shown_pw_id));
193                                } else {
194                                    spawn_fetch(
195                                        self.rt,
196                                        self.client.clone(),
197                                        self.username.clone(),
198                                        self.record_uuid.clone(),
199                                        pw_inbox.sender(),
200                                        PasswordEvent::Show,
201                                    );
202                                }
203                            }
204                        });
205
206                        for (k, v) in self.label.extra() {
207                            if k == "username" {
208                                continue;
209                            }
210                            ui.horizontal(|ui| {
211                                ui.label(egui::RichText::new(format!("{k}:")).strong());
212                                ui.add(egui::Label::new(v.as_str()).truncate());
213                            });
214                        }
215                    });
216            }
217        })
218        .response
219    }
220}

222fn spawn_fetch(
223    rt: &Runtime,
224    client: Arc<EmbeddedHandler>,
225    username: String,
226    record_uuid: Uuid<Record>,
227    tx: egui_inbox::UiInboxSender<PasswordEvent>,
228    wrap: fn(Password) -> PasswordEvent,
229) {
230    rt.spawn(async move {
231        // TODO: surface these errors in the UI instead of stderr.
232        let record = match client
233            .call(Fetch {
234                username,
235                uuid: record_uuid.clone(),
236            })
237            .await
238        {
239            Ok(record) => record,
240            Err(e) => {
241                eprintln!("failed to load record {record_uuid}: {e}");
242                return;
243            }
244        };
245        tx.send(wrap(record.password().clone())).ok();
246    });
