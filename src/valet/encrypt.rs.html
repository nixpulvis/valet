9pub(crate) const SALT_SIZE: usize = 16;
10const NONCE_SIZE: usize = 16;
11
12/// Represents some encrypted data.
13#[derive(PartialEq, Eq)]
14pub struct Encrypted {
15    pub(crate) data: Vec<u8>,
16    pub(crate) nonce: Vec<u8>,
17}
18
19/// A key is generated from a user record's salt and thier password.
20///
21/// Aes256 has a 512-bit key size, and achieves 256-bit security.
22//
23// TODO: #6 keys should not be clonable.
24#[derive(PartialEq, Eq, Clone)]
25pub struct Key(AesKey<Aes256SivAead>);
26
27impl Key {
28    pub fn new() -> Self {
29        Key(Aes256SivAead::generate_key(&mut OsRng))
30    }
31
32    // TODO: Zeroize password
33    pub fn from_password(password: String, salt: &[u8]) -> Result<Self, Error> {
34        let argon2 = Argon2::default();
35        let mut output_key_material = [0u8; <Aes256SivAead as KeySizeUser>::KeySize::USIZE];
36        argon2
37            .hash_password_into(password.as_bytes(), salt, &mut output_key_material)
38            .map_err(|e| Error::KeyDerivation(format!("{}", e)))?;
39
40        Ok(Key(AesKey::<Aes256SivAead>::clone_from_slice(
41            &output_key_material,
42        )))
43    }
44
45    pub(crate) fn generate_salt() -> [u8; SALT_SIZE] {
46        let mut salt = [0; SALT_SIZE];
47        OsRng.fill_bytes(&mut salt);
48        salt
49    }
50
51    pub fn encrypt(&self, plaintext: &[u8]) -> Result<Encrypted, Error> {
52        // TODO: Better security will be achieved by storing a unique counter
53        // for this somewhere.
54        let mut nonce_bytes = [0; NONCE_SIZE];
55        OsRng.fill_bytes(&mut nonce_bytes);
56        let nonce = Nonce::from_slice(&nonce_bytes);
57
58        let cipher = Aes256SivAead::new(&self.0);
59        let ciphertext = cipher
60            .encrypt(nonce, plaintext)
61            .map_err(|e| Error::Encryption(format!("{}", e)))?;
62        Ok(Encrypted {
63            data: ciphertext,
64            nonce: nonce.as_slice().into(),
65        })
66    }
67
68    pub fn decrypt(&self, encrypted: &Encrypted) -> Result<Vec<u8>, Error> {
69        let nonce = Nonce::from_slice(&encrypted.nonce);
70        let cipher = Aes256SivAead::new(&self.0);
71        let plaintext = cipher
72            .decrypt(nonce, &encrypted.data[..])
73            .map_err(|e| Error::Decryption(format!("{}", e)))?;
74        Ok(plaintext)
75    }
76}
77
78#[derive(Debug)]
79pub enum Error {
80    KeyDerivation(String),
81    Encryption(String),
82    Decryption(String),
83}
