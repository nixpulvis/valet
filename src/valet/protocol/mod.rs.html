1//! Handler trait, `Call`-based dispatch, and per-protocol client/server types.
2//!
3//! Three things live here:
4//!
5//! * [`Handler`] / [`SendHandler`]: "turns a [`Request`] into a
6//!   [`Response`]". Every concrete per-protocol client implements one
7//!   of these. [`Handler`] is the base; [`SendHandler`] is the
8//!   `Send + Sync` refinement used by native (tokio) callers. The
9//!   uniform typed-call entry points are [`Handler::call`] and
10//!   [`SendHandler::call`]: callers build a payload struct (e.g.
11//!   [`message::Unlock`]) that implements [`message::Call`] and hand
12//!   it to `call`.
13//! * [`Serve`]: "runs a server loop that forwards every incoming
14//!   request to a [`SendHandler`]". Implemented by [`SocketServer`]
15//!   and [`NativeMessageServer`].
16//! * The concrete per-protocol types ([`EmbeddedHandler`],
17//!   [`SocketClient`] / [`SocketServer`], [`NativeMessageClient`] /
18//!   [`NativeMessageServer`]). Each is feature-gated; a build without
19//!   the matching `protocol-*` feature doesn't see the type.
20//!
21//! The in-process "embedded" case is a [`Handler`] that owns a SQLite
22//! handle and dispatches requests directly. It is not a "protocol";
23//! nothing frames bytes on the wire.
24//!
25//! [`Request`]: message::Request
26//! [`Response`]: message::Response
27

67/// Anything that can turn a [`Request`] into a [`Response`]. Provider
68/// side of every RPC. Base trait with no `Send`/`Sync` requirement;
69/// works for the browser WASM build's [`NativeMessageClient`] whose JS
70/// `Port` is `!Send`. Native callers want the [`SendHandler`] subtrait
71/// instead, which pins down `Send + Sync` on the implementor and
72/// `+ Send` on every returned future.
73///
74/// [`io::Result`] is the outer return for transport failures (socket
75/// dropped, disk full while writing, ...); application-level failures
76/// (locked user, record not found, bad query) are conveyed as
77/// [`Response::Error`]. In-process handlers never return `Err`; `Err`
78/// only comes from remote-forwarding clients.
79///
80/// Typed calls go through [`Handler::call`], which takes any [`Call`]
81/// payload struct and extracts its declared response type.
82pub trait Handler {
83    /// Dispatch one [`Request`] to its [`Response`]. [`Handler::call`]
84    /// is the typed wrapper built on top of this.
85    fn handle(&self, req: Request) -> impl std::future::Future<Output = io::Result<Response>>;
86
87    /// Issue a typed [`Call`]. Converts the payload into a
88    /// [`Request`], dispatches it through [`Handler::handle`], and
89    /// extracts the declared [`Call::Response`].
90    fn call<C: Call>(&self, c: C) -> impl std::future::Future<Output = Result<C::Response, Error>> {
91        async move { Ok(C::from_response(self.handle(c.into_request()).await?)?) }
92    }
93}

95/// `Send + Sync` refinement of [`Handler`]. Every implementor is also
96/// a [`Handler`] via the blanket below; the difference is that
97/// [`SendHandler::handle`] and [`SendHandler::call`] return
98/// [`Send`] futures, which is what `tokio::spawn` and multi-threaded
99/// [`Serve`] loops require. Every native handler
100/// ([`EmbeddedHandler`], [`SocketClient`], `Arc<H>`) impls this.
101pub trait SendHandler: Send + Sync {
102    /// Dispatch one [`Request`] to its [`Response`], returning a
103    /// [`Send`] future.
104    fn handle(
105        &self,
106        req: Request,
107    ) -> impl std::future::Future<Output = io::Result<Response>> + Send;
108
109    /// Send-bounded typed [`Call`].
110    fn call<C: Call + Send>(
111        &self,
112        c: C,
113    ) -> impl std::future::Future<Output = Result<C::Response, Error>> + Send
114    where
115        C::Response: Send,
116    {
117        async move { Ok(C::from_response(self.handle(c.into_request()).await?)?) }
118    }
119}
120
121// Every `SendHandler` is also a `Handler`; the Send bounds just
122// aren't projected. One-way bridge: `Handler` doesn't imply
123// `SendHandler` because the futures aren't guaranteed Send.
124impl<T: SendHandler + ?Sized> Handler for T {
125    fn handle(&self, req: Request) -> impl std::future::Future<Output = io::Result<Response>> {
126        <Self as SendHandler>::handle(self, req)
127    }
128}
129
130// Blanket impl so `Arc<H>` is itself a `SendHandler`. Lets callers
131// hand an `Arc<EmbeddedHandler>` to a server's `serve` without a
132// wrapper, and lets FFI / daemon / transport layers uniformly work
133// with `H: SendHandler`.
134impl<H: SendHandler + ?Sized> SendHandler for std::sync::Arc<H> {
135    fn handle(
136        &self,
137        req: Request,
138    ) -> impl std::future::Future<Output = io::Result<Response>> + Send {
139        (**self).handle(req)
140    }
141}

143/// A server-side role that accepts connections and dispatches each
144/// incoming [`Request`] through a [`SendHandler`]. Implemented by
145/// [`SocketServer`] and [`NativeMessageServer`].
146#[cfg(feature = "_protocols")]
147pub trait Serve {
148    /// Run the accept/dispatch loop, forwarding every request to
149    /// `handler`. Never returns under normal operation.
150    fn serve<H: SendHandler + 'static>(
151        self,
152        handler: std::sync::Arc<H>,
153    ) -> impl std::future::Future<Output = io::Result<()>> + Send;
154}

156/// Top-level error returned by [`Handler::call`] and
157/// [`SendHandler::call`]. Collapses transport failures and
158/// application-level errors into one shape so callers only need one
159/// `?` chain.
160#[derive(Debug)]
161pub enum Error {
162    /// Transport / IO failure (socket dropped, encode/decode failure,
163    /// etc.).
164    Io(io::Error),
165    /// The remote peer's handler returned [`Response::Error`]; the
166    /// payload is its message.
167    Remote(String),
168    /// The remote peer's handler returned a [`Response`] variant that
169    /// doesn't match the one the typed call expected. Indicates a
170    /// version skew between peers.
171    Unexpected,
172}
173
174impl From<io::Error> for Error {
175    fn from(e: io::Error) -> Self {
176        Error::Io(e)
177    }
178}
179
180impl From<message::ResponseError> for Error {
181    fn from(e: message::ResponseError) -> Self {
182        match e {
183            message::ResponseError::Remote(msg) => Error::Remote(msg),
184            message::ResponseError::UnexpectedResponse => Error::Unexpected,
185        }
186    }
187}
188
189impl std::fmt::Display for Error {
190    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
191        match self {
192            Error::Io(e) => write!(f, "io: {e}"),
193            Error::Remote(msg) => write!(f, "remote: {msg}"),
194            Error::Unexpected => write!(f, "unexpected response variant"),
195        }
196    }
197}
