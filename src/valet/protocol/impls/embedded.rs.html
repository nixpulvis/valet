1//! In-process handler: [`EmbeddedHandler`] owns a SQLite handle plus
2//! the cache of unlocked [`User`] / [`Lot`] keys and answers
3//! [`Request`]s directly. There is no wire protocol here; nothing
4//! frames bytes over a socket.
5//!
6//! [`User`]: crate::user::User
7//! [`Lot`]: crate::Lot

28/// Dispatch-layer errors the embedded handler can raise internally.
29/// Flattened into the [`Response::Error`] message string as it leaves
30/// `handle`; kept as a typed enum here so lookup helpers can signal
31/// specific failure modes without building ad-hoc format strings at
32/// every call site.
33#[derive(Debug)]
34enum Error {
35    /// A uuid present in `State::user_lots` had no matching entry in
36    /// `State::lots`. A process-internal invariant violation rather
37    /// than a caller-visible condition; reaching this is a bug.
38    LotCacheMiss(Uuid<Lot>),
39}
40
41impl std::fmt::Display for Error {
42    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
43        match self {
44            Error::LotCacheMiss(uuid) => write!(f, "lot '{uuid}' missing from cache"),
45        }
46    }
47}
48
49impl From<Error> for String {
50    fn from(e: Error) -> Self {
51        e.to_string()
52    }
53}

55/// Delay applied after a failed [`Request::Unlock`]. Makes credential
56/// guessing noticeably slow without being user-visible on the success
57/// path.
58pub const FAILED_UNLOCK_DELAY_MS: u64 = 750;
59
60/// Cached keys are dropped after this much wall-clock inactivity.
61pub const IDLE_TIMEOUT: Duration = Duration::from_secs(5 * 60);
62
63/// How often the reaper checks the idle window.
64pub const IDLE_CHECK_INTERVAL: Duration = Duration::from_secs(15);
65
66/// Cache state shared between [`EmbeddedHandler`] and its idle
67/// reaper. The SQLite handle plus per-user / per-lot key caches
68/// currently held in memory.
69pub struct State {
70    pub db: Database,
71    pub users: HashMap<String, User>,
72    /// Every lot any unlocked user can access, keyed by lot uuid.
73    /// Shared lots (multiple users with access) occupy one slot, not
74    /// one copy per user, so the live storgit store is shared too.
75    pub lots: HashMap<Uuid<Lot>, Lot>,
76    /// Per-user access list into [`lots`](Self::lots). Synced from
77    /// SQLite at specific points (unlock, register, create_lot,
78    /// delete_lot); an entry with an empty `Vec` is a populated
79    /// no-lots state.
80    pub user_lots: HashMap<String, Vec<Uuid<Lot>>>,
81    /// Set whenever there are unlocked users; cleared when state is
82    /// dropped.
83    pub last_activity: Option<Instant>,
84}
85
86impl State {
87    pub fn new(db: Database) -> Self {
88        Self {
89            db,
90            users: HashMap::new(),
91            lots: HashMap::new(),
92            user_lots: HashMap::new(),
93            last_activity: None,
94        }
95    }
96
97    pub fn drop_user(&mut self, username: &str) {
98        self.users.remove(username);
99        self.user_lots.remove(username);
100        self.reap_unreferenced_lots();
101        if self.users.is_empty() {
102            self.last_activity = None;
103        }
104    }
105
106    pub fn drop_all(&mut self) {
107        self.users.clear();
108        self.user_lots.clear();
109        self.lots.clear();
110        self.last_activity = None;
111    }
112
113    /// Drop any lot no unlocked user references. Called after
114    /// `drop_user` so a shared lot stays cached as long as at least
115    /// one accessor is still unlocked.
116    fn reap_unreferenced_lots(&mut self) {
117        let mut live: std::collections::HashSet<Uuid<Lot>> = std::collections::HashSet::new();
118        for uuids in self.user_lots.values() {
119            for uuid in uuids {
120                live.insert(uuid.clone());
121            }
122        }
123        self.lots.retain(|uuid, _| live.contains(uuid));
124    }
125
126    pub fn touch(&mut self) {
127        if !self.users.is_empty() {
128            self.last_activity = Some(Instant::now());
129        }
130    }
131
132    fn get_lot(&self, uuid: &Uuid<Lot>) -> Result<&Lot, Error> {
133        self.lots
134            .get(uuid)
135            .ok_or_else(|| Error::LotCacheMiss(uuid.clone()))
136    }
137
138    /// Cache `lot` and record that `username` has access to it. Used
139    /// after a successful SQLite write to keep `lots` and `user_lots`
140    /// in sync at one call site.
141    fn insert_lot(&mut self, username: &str, lot: Lot) {
142        let uuid = lot.uuid().clone();
143        self.lots.insert(uuid.clone(), lot);
144        self.user_lots
145            .entry(username.to_owned())
146            .or_default()
147            .push(uuid);
148    }
149}

151/// Public handler type. Owns the SQLite handle plus the in-memory
152/// caches of unlocked users and their lots; dispatches every
153/// [`Request`] against that state.
154pub struct EmbeddedHandler {
155    state: Arc<Mutex<State>>,
156}
157
158impl EmbeddedHandler {
159    /// Build a handler around `db` and spawn the idle reaper on `rt`.
160    /// Taking an explicit [`Handle`] makes the "must have a tokio
161    /// runtime" requirement static: the caller cannot name this
162    /// function without naming a live runtime.
163    ///
164    /// [`Handle`]: tokio::runtime::Handle
165    pub fn new(db: Database, rt: &tokio::runtime::Handle) -> Self {
166        let state = Arc::new(Mutex::new(State::new(db)));
167        spawn_reaper(rt, state.clone(), IDLE_TIMEOUT, IDLE_CHECK_INTERVAL);
168        Self { state }
169    }
170
171    /// Open the database at `$VALET_DB` (or [`crate::db::default_url`]
172    /// when unset) and build a handler around it. Used by the `valetd`
173    /// binary and by any transport that just wants the default
174    /// location.
175    pub async fn open_from_env(rt: &tokio::runtime::Handle) -> Result<Self, String> {
176        let db_url = std::env::var("VALET_DB").unwrap_or_else(|_| crate::db::default_url());
177        let db = Database::new(&db_url)
178            .await
179            .map_err(|e| format!("failed to open database at {db_url}: {e:?}"))?;
180        Ok(Self::new(db, rt))
181    }
182}

184impl SendHandler for EmbeddedHandler {
185    async fn handle(&self, req: Request) -> io::Result<Response> {
186        let kind: &'static str = (&req).into();
187        let response = match dispatch(&self.state, req).await {
188            Ok(r) => {
189                let resp_kind: &'static str = (&r).into();
190                info!(request = kind, response = resp_kind, "ok");
191                r
192            }
193            Err(msg) => {
194                warn!(request = kind, "error: {msg}");
195                Response::Error(msg)
196            }
197        };
198        // Any dispatch attempt counts as activity.
199        self.state.lock().await.touch();
200        Ok(response)
201    }
202}

204/// Drop every cached user if the idle window has elapsed since the
205/// last request touched the state. Returns `true` when something was
206/// dropped, so the reaper can log it.
207async fn reap_if_idle(state: &Arc<Mutex<State>>, idle_timeout: Duration) -> bool {
208    let mut st = state.lock().await;
209    match st.last_activity {
210        Some(last) if last.elapsed() >= idle_timeout => {
211            st.drop_all();
212            true
213        }
214        _ => false,
215    }
216}

218fn spawn_reaper(
219    rt: &tokio::runtime::Handle,
220    state: Arc<Mutex<State>>,
221    idle_timeout: Duration,
222    check_interval: Duration,
223) {
224    rt.spawn(async move {
225        info!(
226            idle_timeout_secs = idle_timeout.as_secs(),
227            check_interval_secs = check_interval.as_secs(),
228            "reaper started",
229        );
230        loop {
231            tokio::time::sleep(check_interval).await;
232            if reap_if_idle(&state, idle_timeout).await {
233                info!("idle timeout, locked all users");
234            }
235        }
236    });
237}

239async fn dispatch(state: &Arc<Mutex<State>>, req: Request) -> Result<Response, String> {
240    match req {
241        Request::Status => {
242            let st = state.lock().await;
243            let mut users: Vec<String> = st.users.keys().cloned().collect();
244            users.sort();
245            Ok(Response::Users(users))
246        }
247        Request::ListUsers => {
248            let st = state.lock().await;
249            let users = User::list(&st.db).await.map_err(err)?;
250            Ok(Response::Users(users))
251        }
252        Request::Unlock { username, password } => {
253            let mut st = state.lock().await;
254            match User::load(&st.db, &username, password).await {
255                Ok(user) => {
256                    info!(user = %username, "unlocked");
257                    st.users.insert(username.clone(), user);
258                    sync_user_lots(&mut st, &username).await?;
259                    Ok(Response::Ok)
260                }
261                Err(e) => {
262                    drop(st);
263                    tokio::time::sleep(Duration::from_millis(FAILED_UNLOCK_DELAY_MS)).await;
264                    warn!(user = %username, "unlock failed");
265                    Err(err(e))
266                }
267            }
268        }
269        Request::Lock { username } => {
270            info!(user = %username, "locked");
271            state.lock().await.drop_user(&username);
272            Ok(Response::Ok)
273        }
274        Request::LockAll => {
275            info!("locked all users");
276            state.lock().await.drop_all();
277            Ok(Response::Ok)
278        }
279        Request::List { username, queries } => list(state, &username, &queries).await,
280        Request::Fetch { username, uuid } => fetch_any_lot(state, &username, &uuid).await,
281        Request::FindRecords {
282            username,
283            lot,
284            query,
285        } => find_records(state, &username, &lot, &query).await,
286        Request::GetRecord {
287            username,
288            lot,
289            uuid,
290        } => {
291            let st = state.lock().await;
292            let lot_uuid = lookup_lot_uuid(&st, &username, &lot)?;
293            let l = st.get_lot(&lot_uuid)?;
294            let db = &st.db;
295            let record = Record::show(db, l, &uuid)
296                .await
297                .map_err(err)?
298                .ok_or_else(|| "record not found".to_string())?;
299            Ok(Response::Record(record))
300        }
301        Request::CreateRecord {
302            username,
303            lot,
304            label,
305            password,
306            extra,
307        } => create_record(state, username, lot, label, password, extra).await,
308        Request::GenerateRecord {
309            username,
310            lot,
311            label,
312        } => {
313            let password = Password::generate();
314            let extra = HashMap::new();
315            create_record(state, username, lot, label, password, extra).await
316        }
317        Request::Register { username, password } => {
318            let mut st = state.lock().await;
319            let user = User::new(&username, password)
320                .map_err(err)?
321                .register(&st.db)
322                .await
323                .map_err(err)?;
324            let mut lot = Lot::new(DEFAULT_LOT);
325            lot.save(&st.db, &user).await.map_err(err)?;
326            // Leave the newly-registered user unlocked. The caller has
327            // just proved the password; forcing a follow-up Unlock to
328            // re-derive the key is redundant Argon2 work.
329            st.insert_lot(&username, lot);
330            st.users.insert(username.clone(), user);
331            info!(user = %username, "registered and unlocked");
332            Ok(Response::Ok)
333        }
334        Request::Validate { username, password } => {
335            let st = state.lock().await;
336            let user = match User::load(&st.db, &username, password).await {
337                Ok(u) => u,
338                Err(e) => {
339                    drop(st);
340                    tokio::time::sleep(Duration::from_millis(FAILED_UNLOCK_DELAY_MS)).await;
341                    warn!(user = %username, "validate failed");
342                    return Err(err(e));
343                }
344            };
345            if user.validate() {
346                Ok(Response::Ok)
347            } else {
348                Err("validation token mismatch".to_string())
349            }
350        }
351        Request::ListLots { username } => list_lots(state, &username).await,
352        Request::CreateLot { username, lot } => create_lot(state, &username, &lot).await,
353        Request::DeleteLot { username, lot } => delete_lot(state, &username, &lot).await,
354        Request::History {
355            username,
356            lot,
357            uuid,
358        } => history(state, &username, &lot, &uuid).await,
359    }
360}

362async fn list(
363    state: &Arc<Mutex<State>>,
364    username: &str,
365    queries: &[String],
366) -> Result<Response, String> {
367    use crate::record::Query;
368    let parsed = queries
369        .iter()
370        .map(|s| s.parse::<Query>())
371        .collect::<Result<Vec<_>, _>>()
372        .map_err(|e| format!("invalid query: {e}"))?;
373
374    let st = state.lock().await;
375    let lot_uuids = user_lot_uuids(&st, username)?;
376
377    let mut out: Vec<(Uuid<Record>, Label)> = Vec::new();
378    for lot_uuid in lot_uuids {
379        let lot = st.get_lot(lot_uuid)?;
380        for (label, uuid) in lot.index().iter() {
381            let keep = parsed.is_empty()
382                || parsed
383                    .iter()
384                    .any(|q| q.matches_lot(lot.name()) && q.matches_label(label));
385            if keep {
386                out.push((uuid.clone(), label.clone()));
387            }
388        }
389    }
390    Ok(Response::Index(out))
391}

393async fn fetch_any_lot(
394    state: &Arc<Mutex<State>>,
395    username: &str,
396    uuid: &Uuid<Record>,
397) -> Result<Response, String> {
398    let st = state.lock().await;
399    let lot_uuids = user_lot_uuids(&st, username)?.to_vec();
400    for lot_uuid in lot_uuids {
401        let lot = st.get_lot(&lot_uuid)?;
402        if let Some(record) = Record::show(&st.db, lot, uuid).await.map_err(err)? {
403            return Ok(Response::Record(record));
404        }
405    }
406    Err(format!("no record with uuid {uuid}"))
407}

409async fn find_records(
410    state: &Arc<Mutex<State>>,
411    username: &str,
412    lot: &str,
413    query: &str,
414) -> Result<Response, String> {
415    use crate::protocol::message::label_matches_domain;
416    let st = state.lock().await;
417    let lot_uuid = lookup_lot_uuid(&st, username, lot)?;
418    let l = st.get_lot(&lot_uuid)?;
419    let entries: Vec<(Uuid<Record>, Label)> = l
420        .index()
421        .iter()
422        .filter(|(label, _)| label_matches_domain(label, query))
423        .map(|(label, uuid)| (uuid.clone(), label.clone()))
424        .collect();
425    Ok(Response::Index(entries))
426}

428async fn create_record(
429    state: &Arc<Mutex<State>>,
430    username: String,
431    lot: String,
432    label: Label,
433    password: Password,
434    extra: HashMap<String, String>,
435) -> Result<Response, String> {
436    let mut st = state.lock().await;
437    let lot_uuid = lookup_lot_uuid(&st, &username, &lot)?;
438
439    let mut data = Data::new(password);
440    if !extra.is_empty() {
441        data = data.with_extra(extra);
442    }
443    let State { db, lots, .. } = &mut *st;
444    let l = lots
445        .get_mut(&lot_uuid)
446        .ok_or_else(|| Error::LotCacheMiss(lot_uuid.clone()))?;
447    // Upsert: reuse the uuid already bound to this label name so
448    // storgit extends that submodule's history, rather than minting a
449    // fresh uuid on every save.
450    let record = match l.index().find_by_name(label.name()).cloned() {
451        Some(existing) => Record::with_uuid(existing, l, label, data),
452        None => Record::new(l, label, data),
453    };
454    record.save(db, l).await.map_err(err)?;
455    info!(user = %username, lot = %lot, uuid = %record.uuid(), "record saved");
456    Ok(Response::Record(record))
457}

459async fn list_lots(state: &Arc<Mutex<State>>, username: &str) -> Result<Response, String> {
460    let st = state.lock().await;
461    let lot_uuids = user_lot_uuids(&st, username)?;
462    let mut entries: Vec<(Uuid<Lot>, String)> = lot_uuids
463        .iter()
464        .map(|u| st.get_lot(u).map(|l| (u.clone(), l.name().to_owned())))
465        .collect::<Result<_, _>>()?;
466    entries.sort_by(|a, b| a.1.cmp(&b.1));
467    Ok(Response::Lots(entries))
468}

470async fn create_lot(
471    state: &Arc<Mutex<State>>,
472    username: &str,
473    lot_name: &str,
474) -> Result<Response, String> {
475    let mut st = state.lock().await;
476    let user = st
477        .users
478        .get(username)
479        .ok_or_else(|| format!("user '{username}' is locked"))?;
480    let mut lot = Lot::new(lot_name);
481    lot.save(&st.db, user).await.map_err(err)?;
482    info!(user = %username, lot = %lot_name, "lot created");
483    st.insert_lot(username, lot);
484    Ok(Response::Ok)
485}

487async fn delete_lot(
488    state: &Arc<Mutex<State>>,
489    username: &str,
490    lot_name: &str,
491) -> Result<Response, String> {
492    let mut st = state.lock().await;
493    let lot_uuid = lookup_lot_uuid(&st, username, lot_name)?;
494    // Lot rows cascade-delete in SQLite, so access is revoked for
495    // every user regardless of who initiated. Mirror that: drop the
496    // lot from every cache entry, not just this user's.
497    for uuids in st.user_lots.values_mut() {
498        uuids.retain(|u| u != &lot_uuid);
499    }
500    let lot = st
501        .lots
502        .remove(&lot_uuid)
503        .ok_or_else(|| Error::LotCacheMiss(lot_uuid.clone()))?;
504    lot.delete(&st.db).await.map_err(err)?;
505    info!(user = %username, lot = %lot_name, "lot deleted");
506    Ok(Response::Ok)
507}

509async fn history(
510    state: &Arc<Mutex<State>>,
511    username: &str,
512    lot_name: &str,
513    uuid: &Uuid<Record>,
514) -> Result<Response, String> {
515    let st = state.lock().await;
516    let lot_uuid = lookup_lot_uuid(&st, username, lot_name)?;
517    let l = st.get_lot(&lot_uuid)?;
518    let revisions = Record::history(&st.db, l, uuid)
519        .await
520        .map_err(err)?
521        .ok_or_else(|| format!("record '{uuid}' not found in lot '{lot_name}'"))?;
522    let entries = revisions
523        .into_iter()
524        .map(|rev| {
525            let time_millis = rev
526                .time
527                .duration_since(std::time::UNIX_EPOCH)
528                .map(|d| d.as_millis() as i64)
529                .unwrap_or_else(|e| -(e.duration().as_millis() as i64));
530            let password = rev.data.password().clone();
531            RevisionEntry {
532                time_millis,
533                label: rev.label,
534                password,
535            }
536        })
537        .collect();
538    Ok(Response::History(entries))
539}

541/// Eager-sync the cached lot set for `username` against SQLite. Loads
542/// every lot the user has access to, de-duplicates by uuid into
543/// [`State::lots`], and records the user's uuid list in
544/// [`State::user_lots`]. Called at the boundary events that change lot
545/// membership (unlock, create_lot, delete_lot) rather than on every
546/// cross-lot read.
547async fn sync_user_lots(st: &mut State, username: &str) -> Result<(), String> {
548    let user = st
549        .users
550        .get(username)
551        .ok_or_else(|| format!("user '{username}' is locked"))?;
552    let lots = Lot::load_all(&st.db, user).await.map_err(err)?;
553    let mut uuids = Vec::with_capacity(lots.len());
554    for lot in lots {
555        let uuid = lot.uuid().clone();
556        uuids.push(uuid.clone());
557        // First user to load wins; subsequent identical lots (shared
558        // across users) reuse the already-cached entry with its live
559        // storgit store.
560        st.lots.entry(uuid).or_insert(lot);
561    }
562    st.user_lots.insert(username.to_owned(), uuids);
563    Ok(())
564}

566/// Look up the uuid of `lot_name` for `username`, reading only from
567/// cache (no DB). Errors if the user is locked or the lot isn't in
568/// their access list.
569fn lookup_lot_uuid(st: &State, username: &str, lot_name: &str) -> Result<Uuid<Lot>, String> {
570    let uuids = user_lot_uuids(st, username)?;
571    for uuid in uuids {
572        if st.get_lot(uuid)?.name() == lot_name {
573            return Ok(uuid.clone());
574        }
575    }
576    Err(format!("lot '{lot_name}' not found"))
577}
578
579/// Borrow the per-user uuid list. Errors if the user isn't unlocked.
580fn user_lot_uuids<'a>(st: &'a State, username: &str) -> Result<&'a [Uuid<Lot>], String> {
581    if !st.users.contains_key(username) {
582        return Err(format!("user '{username}' is locked"));
583    }
584    Ok(st
585        .user_lots
586        .get(username)
587        .map(|v| v.as_slice())
588        .unwrap_or(&[]))
589}
