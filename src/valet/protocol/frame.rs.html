1//! Wire framing for [`Request`] and [`Response`].
2//!
3//! Both forms share the same bitcode body; they differ only in the
4//! outer envelope each transport expects:
5//!
6//! * Length-prefixed (4-byte big-endian length, then payload). The
7//!   sync [`Frame::send`]/[`Frame::recv`] compile everywhere; the
8//!   [`Frame::send_async`]/[`Frame::recv_async`] tokio variants are
9//!   gated behind the private `_async-io` feature so a
10//!   wasm-friendly consumer (the browser extension) can depend on
11//!   [`Request`]/[`Response`] without pulling in tokio.
12//! * Base64 of the bitcode body, for stuffing inside the browser
13//!   native-messaging JSON envelope.
14//!
15//! [`Request`]: super::message::Request
16//! [`Response`]: super::message::Response
17
18use super::message::{Request, Response};
19use bitcode::{Decode, Encode};
20use std::io::{self, Read, Write};
21#[cfg(feature = "_async-io")]
22use tokio::io::{AsyncRead, AsyncReadExt, AsyncWrite, AsyncWriteExt};
23
24/// Maximum allowed frame payload, 16 MiB. The daemon never returns anywhere
25/// near this much data; the cap exists to bound client-side allocations if
26/// a peer misbehaves.
27pub const MAX_FRAME_LEN: usize = 16 * 1024 * 1024;
28
29fn check_len(len: usize) -> io::Result<()> {
30    if len > MAX_FRAME_LEN {
31        Err(io::Error::new(
32            io::ErrorKind::InvalidData,
33            "frame exceeds MAX_FRAME_LEN",
34        ))
35    } else {
36        Ok(())
37    }
38}
39
40fn write_frame<W: Write>(w: &mut W, payload: &[u8]) -> io::Result<()> {
41    check_len(payload.len())?;
42    let len = u32::try_from(payload.len()).expect("checked above");
43    w.write_all(&len.to_be_bytes())?;
44    w.write_all(payload)?;
45    w.flush()
46}
47
48fn read_frame<R: Read>(r: &mut R) -> io::Result<Vec<u8>> {
49    let mut len_bytes = [0u8; 4];
50    r.read_exact(&mut len_bytes)?;
51    let len = u32::from_be_bytes(len_bytes) as usize;
52    check_len(len)?;
53    let mut buf = vec![0u8; len];
54    r.read_exact(&mut buf)?;
55    Ok(buf)
56}
57
58/// Write one length-prefixed frame of already-encoded bytes to `w`. The
59/// payload is anything up to [`MAX_FRAME_LEN`] bytes; callers that want
60/// the typed encode-then-send should use [`Frame::send_async`] instead.
61#[cfg(feature = "_async-io")]
62pub async fn send_frame_async<W: AsyncWrite + Unpin>(w: &mut W, payload: &[u8]) -> io::Result<()> {
63    check_len(payload.len())?;
64    let len = u32::try_from(payload.len()).expect("checked above");
65    w.write_all(&len.to_be_bytes()).await?;
66    w.write_all(payload).await?;
67    w.flush().await
68}
69
70/// Read one length-prefixed frame from `r`. Inverse of [`send_frame_async`].
71#[cfg(feature = "_async-io")]
72pub async fn recv_frame_async<R: AsyncRead + Unpin>(r: &mut R) -> io::Result<Vec<u8>> {
73    let mut len_bytes = [0u8; 4];
74    r.read_exact(&mut len_bytes).await?;
75    let len = u32::from_be_bytes(len_bytes) as usize;
76    check_len(len)?;
77    let mut buf = vec![0u8; len];
78    r.read_exact(&mut buf).await?;
79    Ok(buf)
80}
81
82fn decode_err<E: std::fmt::Display>(e: E) -> io::Error {
83    io::Error::new(io::ErrorKind::InvalidData, format!("bitcode decode: {e}"))
84}

86/// Shared framing for wire messages. The length-prefixed `send`/`recv`
87/// helpers speak the Unix-socket wire format used between the daemon and
88/// its remote clients. The `encode_base64`/`decode_base64` helpers speak
89/// the envelope the browser native-messaging shim stuffs inside its JSON
90/// frames; same bitcode payload, different outer wrapper.
91///
92/// TODO: before a proper release, add a version discriminator (one magic
93/// byte or a leading u16) to both the length-prefixed and base64 forms.
94/// Bitcode enum tags are positional, so adding or removing a `Request` /
95/// `Response` variant silently misdecodes across mismatched peers today.
96pub trait Frame: Encode + for<'de> Decode<'de> + Sized {
97    /// Bitcode-encode `self` into a freshly-allocated buffer, without any
98    /// framing. The length-prefix and base64 helpers below are built on
99    /// top of this; callers that already have their own framing (the
100    /// browser native-messaging shim's embedded mode) use it directly.
101    fn encode(&self) -> Vec<u8> {
102        bitcode::encode(self)
103    }
104
105    /// Inverse of [`encode`](Self::encode). Decode failures are surfaced
106    /// as `io::Error` with kind [`io::ErrorKind::InvalidData`].
107    fn decode(bytes: &[u8]) -> io::Result<Self> {
108        bitcode::decode(bytes).map_err(decode_err)
109    }
110
111    /// Bitcode-encode `self` and write it as one length-prefixed frame.
112    fn send<W: Write>(&self, w: &mut W) -> io::Result<()> {
113        write_frame(w, &self.encode())
114    }
115
116    /// Read one length-prefixed frame and bitcode-decode it.
117    fn recv<R: Read>(r: &mut R) -> io::Result<Self> {
118        Self::decode(&read_frame(r)?)
119    }
120
121    /// Async [`send`](Self::send) over a tokio writer.
122    #[cfg(feature = "_async-io")]
123    fn send_async<W: AsyncWrite + Unpin + Send>(
124        &self,
125        w: &mut W,
126    ) -> impl std::future::Future<Output = io::Result<()>> + Send
127    where
128        Self: Sync,
129    {
130        async move { send_frame_async(w, &self.encode()).await }
131    }
132
133    /// Async [`recv`](Self::recv) over a tokio reader.
134    #[cfg(feature = "_async-io")]
135    fn recv_async<R: AsyncRead + Unpin + Send>(
136        r: &mut R,
137    ) -> impl std::future::Future<Output = io::Result<Self>> + Send {
138        async move { Self::decode(&recv_frame_async(r).await?) }
139    }
140
141    /// Bitcode-encode `self` and base64 it, for embedding in the browser
142    /// native-messaging JSON envelope.
143    fn encode_base64(&self) -> String {
144        use base64::{Engine, engine::general_purpose::STANDARD};
145        STANDARD.encode(self.encode())
146    }
147
148    /// Inverse of [`encode_base64`](Self::encode_base64).
149    fn decode_base64(b64: &str) -> Result<Self, DecodeError> {
150        use base64::{Engine, engine::general_purpose::STANDARD};
151        let bytes = STANDARD.decode(b64).map_err(DecodeError::Base64)?;
152        bitcode::decode(&bytes).map_err(DecodeError::Bitcode)
153    }
154}
155
156impl Frame for Request {}
157impl Frame for Response {}
158
159/// Errors from [`Frame::decode_base64`].
160#[derive(Debug)]
161pub enum DecodeError {
162    /// The base64 envelope was malformed.
163    Base64(base64::DecodeError),
164    /// The bitcode payload could not be decoded.
165    Bitcode(bitcode::Error),
166}
167
168impl std::fmt::Display for DecodeError {
169    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
170        match self {
171            DecodeError::Base64(e) => write!(f, "base64: {e}"),
172            DecodeError::Bitcode(e) => write!(f, "bitcode: {e}"),
173        }
174    }
175}
176
177impl std::error::Error for DecodeError {}

201    #[test]
202    fn request_round_trip_list() {
203        let (mut a, mut b) = UnixStream::pair().unwrap();
204        let sent = Request::List {
205            username: "alice".into(),
206            queries: vec!["github.com".into(), "example.com".into()],
207        };
208        sent.send(&mut a).unwrap();
209        let got = Request::recv(&mut b).unwrap();
210        match got {
211            Request::List { username, queries } => {
212                assert_eq!(username, "alice");
213                assert_eq!(queries, vec!["github.com", "example.com"]);
214            }
215            _ => panic!("wrong variant"),
216        }
217    }

221        let (mut a, mut b) = UnixStream::pair().unwrap();
222        let sent = Request::Unlock {

224            password: "s3cret!!".try_into().unwrap(),
225        };
226        sent.send(&mut a).unwrap();
227        let got = Request::recv(&mut b).unwrap();
