15/// A message sent from a client to the handler. Each variant is answered by
16/// exactly one [`Response`] (possibly [`Response::Error`]).
17#[derive(Encode, Decode, Debug, strum::IntoStaticStr)]
18#[strum(serialize_all = "snake_case")]
19pub enum Request {
20    /// List currently unlocked usernames. Answered with [`Response::Users`].
21    Status,
22    /// List every registered username. Answered with [`Response::Users`].
23    ListUsers,
24    /// Derive the user's key and cache the unlocked [`crate::user::User`] on
25    /// the handler. Answered with [`Response::Ok`] or [`Response::Error`].
26    Unlock {
27        username: String,
28        password: Password,
29    },
30    /// Drop the cached keys for one user. Answered with [`Response::Ok`].
31    Lock { username: String },
32    /// Drop every cached user. Answered with [`Response::Ok`].
33    LockAll,
34    /// Cross-lot query-language search. `queries` are parsed as
35    /// [`crate::record::Query`]; empty means every record in every lot this
36    /// user has access to. Answered with [`Response::Index`].
37    List {
38        username: String,
39        queries: Vec<String>,
40    },
41    /// Fetch one decrypted [`Record`] by uuid. The handler searches the
42    /// user's lots until it finds a match. Answered with [`Response::Record`].
43    Fetch {
44        username: String,
45        uuid: Uuid<Record>,
46    },
47    // TODO: fold this into `List` by adding a `Query::Domain` variant to
48    // `valet::record::Query` that carries the symmetric-suffix match
49    // semantics. Then `FindRecords`, `domain_matches`, and
50    // `label_matches_domain` can all go away.
51    /// Per-lot domain match. `query` is a host string; the handler returns
52    /// the label-and-uuid pairs of every matching record, without
53    /// decrypting passwords. Answered with [`Response::Index`].
54    FindRecords {
55        username: String,
56        lot: String,
57        query: String,
58    },
59    /// Fetch one decrypted [`Record`] by uuid in a specific lot. Answered
60    /// with [`Response::Record`].
61    GetRecord {
62        username: String,
63        lot: String,
64        uuid: Uuid<Record>,
65    },
66    /// Create a new record with a caller-supplied password. Answered with
67    /// [`Response::Record`] carrying the stored record.
68    CreateRecord {
69        username: String,
70        lot: String,
71        label: Label,
72        password: Password,
73        extra: HashMap<String, String>,
74    },
75    /// Create a new record with a handler-generated password. Answered with
76    /// [`Response::Record`] carrying the stored record.
77    GenerateRecord {
78        username: String,
79        lot: String,
80        label: Label,
81    },
82    /// Register a new user and their default lot. Answered with
83    /// [`Response::Ok`] or [`Response::Error`].
84    Register {
85        username: String,
86        password: Password,
87    },
88    /// Verify a password against a user's stored salt + validation token
89    /// without caching the unlocked state. Answered with [`Response::Ok`]
90    /// or [`Response::Error`].
91    Validate {
92        username: String,
93        password: Password,
94    },
95    /// List every lot the user has access to. Answered with
96    /// [`Response::Lots`].
97    ListLots { username: String },
98    /// Create a new lot owned by the user. Answered with [`Response::Ok`].
99    CreateLot { username: String, lot: String },
100    /// Delete a lot. Answered with [`Response::Ok`].
101    DeleteLot { username: String, lot: String },
102    /// Walk the record's historical revisions (newest first). Answered
103    /// with [`Response::History`].
104    History {
105        username: String,
106        lot: String,
107        uuid: Uuid<Record>,
108    },
109}

111/// One historical revision of a record, as carried in
112/// [`Response::History`]. A wire-friendly subset of the in-process
113/// [`crate::record::Revision`] (no `storgit::CommitId`, no `SystemTime`
114/// baggage).
115#[derive(Encode, Decode, Debug)]
116pub struct RevisionEntry {
117    /// Commit timestamp, milliseconds since the Unix epoch. Negative
118    /// for pre-1970 commits (clock skew).
119    pub time_millis: i64,
120    pub label: Label,
121    pub password: Password,
122}

124/// A message sent from the handler back to the client in reply to a
125/// [`Request`]. Every request produces exactly one response; [`Error`] may
126/// be returned in place of any success variant.
127///
128/// [`Error`]: Response::Error
129#[derive(Encode, Decode, Debug, strum::IntoStaticStr)]
130#[strum(serialize_all = "snake_case")]
131pub enum Response {
132    /// Generic success with no payload (Unlock, Lock, LockAll).
133    Ok,
134    /// Username list (Status, ListUsers).
135    Users(Vec<String>),
136    /// Label-and-uuid pairs for every matching record. No password material
137    /// crosses the wire. Answered by [`Request::List`] and
138    /// [`Request::FindRecords`].
139    Index(Vec<(Uuid<Record>, Label)>),
140    /// A single decrypted record (Fetch, GetRecord, CreateRecord,
141    /// GenerateRecord).
142    Record(Record),
143    /// Lot uuid plus name list (ListLots). Sorted by name.
144    Lots(Vec<(Uuid<Lot>, String)>),
145    /// Record-revision list, newest first (History).
146    History(Vec<RevisionEntry>),
147    /// Human-readable error message, returned in place of any success variant
148    /// when the handler cannot satisfy the request.
149    // TODO: Make a proper Error enum for this too
150    Error(String),
151}

153// TODO: goes away with the `Request::FindRecords` fold (see the TODO on
154// that variant above).
155/// Return `true` if a domain label matches a query host, using the loose
156/// suffix rule that powers the [`Request::FindRecords`] RPC. The match is
157/// symmetric across the dot boundary: `"sub.example.com"` matches
158/// `"example.com"` and vice-versa; `"example.com"` does not match
159/// `"other.com"`.
160pub fn domain_matches(record_domain: &str, query: &str) -> bool {
161    let r = record_domain.to_lowercase();
162    let q = query.to_lowercase();
163    r == q || q.ends_with(&format!(".{r}")) || r.ends_with(&format!(".{q}"))
164}
165
166// TODO: goes away with the `Request::FindRecords` fold (see the TODO on
167// that variant above).
168/// Return `true` if `label` is a domain label matching `query` under
169/// [`domain_matches`].
170pub fn label_matches_domain(label: &Label, query: &str) -> bool {
171    match label.name() {
172        LabelName::Domain { domain, .. } => domain_matches(domain, query),
173        _ => false,
174    }
175}

177/// RPC-layer error for valet handler clients. `T` is the transport-specific
178/// failure type: [`std::io::Error`] for the socket handler and FFI, a richer
179/// enum for the browser extension's native-messaging path.
180///
181/// [`Rpc`](Self::Rpc) means no application-level reply arrived
182/// (transport failure, decode failure, disconnect).
183/// [`Response`](Self::Response) carries a reply-inspection failure
184/// ([`ResponseError`]): either a peer-reported error or an unexpected
185/// response variant.
186#[derive(Debug)]
187pub enum Error<T> {
188    /// The RPC round-trip itself failed.
189    Rpc(T),
190    /// A valid reply arrived but could not be interpreted as the
191    /// expected [`Response`] variant.
192    Response(ResponseError),
193}
194
195impl<T: std::fmt::Display> std::fmt::Display for Error<T> {
196    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
197        match self {
198            Error::Rpc(e) => write!(f, "rpc: {e}"),
199            Error::Response(r) => write!(f, "{r}"),
200        }
201    }
202}
203
204impl<T: std::fmt::Debug + std::fmt::Display> std::error::Error for Error<T> {}

206/// Failure variants produced when extracting an expected [`Response`]
207/// variant via the `Response::expect_*` methods. Converts into
208/// [`Error`] for any `T` via `?`.
209#[derive(Debug)]
210pub enum ResponseError {
211    /// The peer's server-side handler returned a failure; the string is
212    /// the message it put inside [`Response::Error`].
213    Remote(String),
214    /// The peer returned a valid [`Response`] of the wrong variant.
215    UnexpectedResponse,
216}
217
218impl<T> From<ResponseError> for Error<T> {
219    fn from(e: ResponseError) -> Self {
220        Error::Response(e)
221    }
222}
223
224impl std::fmt::Display for ResponseError {
225    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
226        match self {
227            ResponseError::Remote(msg) => write!(f, "remote: {msg}"),
228            ResponseError::UnexpectedResponse => write!(f, "unexpected response variant"),
229        }
230    }
231}
232
233impl std::error::Error for ResponseError {}
234
235impl Response {
236    /// Extract [`Response::Ok`]. Folds [`Response::Error`] and
237    /// any other variant into [`ResponseError`].
238    pub fn expect_ok(self) -> Result<(), ResponseError> {
239        match self {
240            Response::Ok => Ok(()),
241            Response::Error(msg) => Err(ResponseError::Remote(msg)),
242            _ => Err(ResponseError::UnexpectedResponse),
243        }
244    }
245
246    /// Extract [`Response::Users`]. Folds [`Response::Error`] and any
247    /// other variant into [`ResponseError`].
248    pub fn expect_users(self) -> Result<Vec<String>, ResponseError> {
249        match self {
250            Response::Users(v) => Ok(v),
251            Response::Error(msg) => Err(ResponseError::Remote(msg)),
252            _ => Err(ResponseError::UnexpectedResponse),
253        }
254    }
255
256    /// Extract [`Response::Index`]. Folds [`Response::Error`] and any
257    /// other variant into [`ResponseError`].
258    pub fn expect_index(self) -> Result<Vec<(Uuid<Record>, Label)>, ResponseError> {
259        match self {
260            Response::Index(v) => Ok(v),
261            Response::Error(msg) => Err(ResponseError::Remote(msg)),
262            _ => Err(ResponseError::UnexpectedResponse),
263        }
264    }
265
266    /// Extract [`Response::Record`]. Folds [`Response::Error`] and any
267    /// other variant into [`ResponseError`].
268    pub fn expect_record(self) -> Result<Record, ResponseError> {
269        match self {
270            Response::Record(r) => Ok(r),
271            Response::Error(msg) => Err(ResponseError::Remote(msg)),
272            _ => Err(ResponseError::UnexpectedResponse),
273        }
274    }
275
276    /// Extract [`Response::Lots`]. Folds [`Response::Error`] and any
277    /// other variant into [`ResponseError`].
278    pub fn expect_lots(self) -> Result<Vec<(Uuid<Lot>, String)>, ResponseError> {
279        match self {
280            Response::Lots(v) => Ok(v),
281            Response::Error(msg) => Err(ResponseError::Remote(msg)),
282            _ => Err(ResponseError::UnexpectedResponse),
283        }
284    }
285
286    /// Extract [`Response::History`]. Folds [`Response::Error`] and any
287    /// other variant into [`ResponseError`].
288    pub fn expect_history(self) -> Result<Vec<RevisionEntry>, ResponseError> {
289        match self {
290            Response::History(v) => Ok(v),
291            Response::Error(msg) => Err(ResponseError::Remote(msg)),
292            _ => Err(ResponseError::UnexpectedResponse),
293        }
294    }
295}
