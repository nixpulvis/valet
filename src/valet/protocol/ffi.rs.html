1//! C ABI surface over the typed [`Handler::call`] API.
2//!
3//! Structure:
4//!
5//! * [`ValetClient`] - opaque handle holding a concrete [`Handler`]
6//!   plus a tokio runtime to block on its async methods. The handler
7//!   type is feature-selected at compile time ([`EmbeddedHandler`]
8//!   with `protocol-embedded`, otherwise [`SocketClient`]); exactly
9//!   one constructor is visible per build.
10//! * Per-protocol constructors: `valet_ffi_client_new_embedded`,
11//!   `valet_ffi_client_new_socket`. Each is gated on its protocol
12//!   feature; with both enabled, `protocol-embedded` wins by
13//!   precedence.
14//! * Per-method shims: `valet_ffi_client_status`, `_unlock`, `_list`,
15//!   `_find_records`, `_fetch`, `_list_users`, `_free`. These don't
16//!   care which handler the handle wraps; they issue typed [`Call`]s
17//!   through the shared [`Handler::call`] surface.
18//!
19//! Shared types (`ValetStr`, `ValetRecordIndex`, `ValetRecord`,
20//! `VALET_FFI_*` return codes, `valet_ffi_last_error`,
21//! `valet_ffi_record_index_free`, `valet_ffi_record_free`) come from
22//! [`crate::ffi`] so clients never link two diverging copies.
23//!
24//! [`Handler`]: super::Handler
25//! [`Handler::call`]: super::Handler::call
26//! [`Call`]: crate::protocol::message::Call
27//! [`EmbeddedHandler`]: crate::protocol::EmbeddedHandler
28//! [`SocketClient`]: crate::protocol::SocketClient
29

47/// The concrete handler the linked staticlib wraps. Feature-selected
48/// at compile time: with `protocol-embedded` the handle carries an
49/// in-proc local handler; otherwise with `protocol-socket` it
50/// carries a socket client.
51#[cfg(feature = "protocol-embedded")]
52type Inner = EmbeddedHandler;
53#[cfg(all(feature = "protocol-socket", not(feature = "protocol-embedded")))]
54type Inner = SocketClient;

56/// Opaque FFI handle. The C side only ever sees `*mut ValetClient`.
57///
58/// Owns a multi-threaded tokio runtime used to drive the async
59/// handler methods to completion synchronously - the C ABI is
60/// blocking. The inner handler takes `&self` and does its own locking,
61/// so concurrent C callers are serialized correctly without a second
62/// mutex here.
63pub struct ValetClient {
64    inner: Inner,
65    rt: Runtime,
66}
67
68/// Error variants produced inside a `valet_ffi_client_*` wrapper.
69#[derive(Debug)]
70enum FfiCallError {
71    Null(&'static str),
72    InvalidUtf8(&'static str),
73    InvalidUuid(&'static str),
74    PasswordTooLong,
75    Io(io::Error),
76    Remote(String),
77    Unexpected,
78}
79
80impl From<io::Error> for FfiCallError {
81    fn from(e: io::Error) -> Self {
82        FfiCallError::Io(e)
83    }
84}
85
86impl From<ClientError> for FfiCallError {
87    fn from(e: ClientError) -> Self {
88        match e {
89            ClientError::Io(e) => FfiCallError::Io(e),
90            ClientError::Remote(msg) => FfiCallError::Remote(msg),
91            ClientError::Unexpected => FfiCallError::Unexpected,
92        }
93    }
94}
95
96impl FfiError for FfiCallError {
97    fn code(&self) -> i32 {
98        match self {
99            FfiCallError::Null(_) => VALET_FFI_ERR_NULL_ARG,
100            FfiCallError::InvalidUtf8(_) => VALET_FFI_ERR_INVALID_UTF8,
101            FfiCallError::InvalidUuid(_) => VALET_FFI_ERR_INVALID_ARG,
102            FfiCallError::PasswordTooLong => VALET_FFI_ERR_PASSWORD_TOO_LONG,
103            FfiCallError::Io(_) => VALET_FFI_ERR_IO,
104            FfiCallError::Remote(_) | FfiCallError::Unexpected => VALET_FFI_ERR_PROTOCOL,
105        }
106    }
107
108    fn message(&self) -> String {
109        match self {
110            FfiCallError::Null(what) => format!("{what} is null"),
111            FfiCallError::InvalidUtf8(what) => format!("{what} is not valid UTF-8"),
112            FfiCallError::InvalidUuid(what) => format!("{what} is not a valid UUID"),
113            FfiCallError::PasswordTooLong => "password too long".into(),
114            FfiCallError::Io(e) => format!("io: {e}"),
115            FfiCallError::Remote(msg) => format!("remote: {msg}"),
116            FfiCallError::Unexpected => "unexpected response variant".into(),
117        }
118    }
119}

121fn not_null<T>(p: *const T, name: &'static str) -> Result<(), FfiCallError> {
122    if p.is_null() {
123        Err(FfiCallError::Null(name))
124    } else {
125        Ok(())
126    }
127}

129/// # Safety
130///
131/// `p` must be non-null and point to a null-terminated C string valid
132/// for reads up to and including the terminator.
133unsafe fn cstr_to_string(p: *const c_char, name: &'static str) -> Result<String, FfiCallError> {
134    unsafe { CStr::from_ptr(p) }
135        .to_str()
136        .map(str::to_owned)
137        .map_err(|_| FfiCallError::InvalidUtf8(name))
138}
139
140/// # Safety
141///
142/// If `len > 0`, `p` must be non-null and point to `len` initialized
143/// bytes. If `len == 0`, `p` may be null.
144unsafe fn bytes_to_string(
145    p: *const c_char,
146    len: usize,
147    name: &'static str,
148) -> Result<String, FfiCallError> {
149    let bytes = if len == 0 {
150        &[][..]
151    } else {
152        unsafe { slice::from_raw_parts(p as *const u8, len) }
153    };
154    std::str::from_utf8(bytes)
155        .map(str::to_owned)
156        .map_err(|_| FfiCallError::InvalidUtf8(name))
157}

159/// Borrow a `ValetClient` from a raw pointer, null-checked.
160unsafe fn borrow<'a>(client: *mut ValetClient) -> Result<&'a ValetClient, FfiCallError> {
161    unsafe { client.as_ref() }.ok_or(FfiCallError::Null("client"))
162}

164fn new_runtime() -> io::Result<Runtime> {
165    // Multi-thread so callers can invoke valet's storgit work, which
166    // uses `tokio::task::block_in_place` internally for the DB-backed
167    // fetcher. `block_in_place` panics on a `current_thread` runtime.
168    // A small worker count keeps the cost low for the FFI use case
169    // (one in-process call at a time).
170    tokio::runtime::Builder::new_multi_thread()
171        .worker_threads(2)
172        .enable_all()
173        .build()
174}

176/// # Safety
177///
178/// `db_path` must point to a null-terminated UTF-8 C string. `out`
179/// must be non-null and writable. On success `*out` holds a freshly
180/// boxed [`ValetClient`] that the caller frees with
181/// [`valet_ffi_client_free`].
182#[cfg(feature = "protocol-embedded")]
183#[unsafe(no_mangle)]
184pub unsafe extern "C" fn valet_ffi_client_new_embedded(
185    db_path: *const c_char,
186    out: *mut *mut ValetClient,
187) -> i32 {
188    ffi::report((|| -> Result<(), FfiCallError> {
189        not_null(db_path, "db_path")?;
190        not_null(out, "out")?;
191        let path = unsafe { cstr_to_string(db_path, "db_path") }?;
192        let rt = new_runtime()?;
193        let db = rt
194            .block_on(crate::db::Database::new(&path))
195            .map_err(|e| FfiCallError::Io(io::Error::other(format!("{e:?}"))))?;
196        let inner = EmbeddedHandler::new(db, rt.handle());
197        let boxed = Box::new(ValetClient { inner, rt });
198        unsafe { *out = Box::into_raw(boxed) };
199        Ok(())
200    })())
201}

227/// # Safety
228///
229/// `client` must be a valid pointer previously returned by one of the
230/// `valet_ffi_client_new_*` constructors and not yet freed. `username`
231/// must point to a null-terminated UTF-8 C string. If `password_len > 0`,
232/// `password` must point to `password_len` initialized bytes; if
233/// `password_len == 0`, `password` may be null.
234#[unsafe(no_mangle)]
235pub unsafe extern "C" fn valet_ffi_client_unlock(
236    client: *mut ValetClient,
237    username: *const c_char,
238    password: *const c_char,
239    password_len: usize,
240) -> i32 {
241    ffi::report((|| -> Result<(), FfiCallError> {
242        let client = unsafe { borrow(client) }?;
243        not_null(username, "username")?;
244        if password.is_null() && password_len > 0 {
245            return Err(FfiCallError::Null("password"));
246        }
247        let username = unsafe { cstr_to_string(username, "username") }?;
248        let pw_str = unsafe { bytes_to_string(password, password_len, "password") }?;
249        let pw: Password = pw_str
250            .as_str()
251            .try_into()
252            .map_err(|_| FfiCallError::PasswordTooLong)?;
253        client.rt.block_on(client.inner.call(Unlock {
254            username,
255            password: pw,
256        }))?;
257        Ok(())
258    })())
259}

299/// # Safety
300///
301/// `client` must be a valid pointer previously returned by one of the
302/// `valet_ffi_client_new_*` constructors and not yet freed.
303/// `username` must point to a null-terminated UTF-8 C string. If
304/// `queries_count > 0`, `queries` and `query_lens` must each point to
305/// `queries_count` initialized elements, every non-null `queries[i]`
306/// valid for `query_lens[i]` bytes. `out` must be non-null and
307/// writable; on success it receives a heap-allocated
308/// [`ValetRecordIndex`] that the caller must release with
309/// [`crate::ffi::valet_ffi_record_index_free`].
310#[unsafe(no_mangle)]
311pub unsafe extern "C" fn valet_ffi_client_list(
312    client: *mut ValetClient,
313    username: *const c_char,
314    queries: *const *const c_char,
315    query_lens: *const usize,
316    queries_count: usize,
317    out: *mut ValetRecordIndex,
318) -> i32 {
319    ffi::report((|| -> Result<(), FfiCallError> {
320        let client = unsafe { borrow(client) }?;
321        not_null(username, "username")?;
322        not_null(out, "out")?;
323        let username = unsafe { cstr_to_string(username, "username") }?;
324        let queries = unsafe { collect_queries(queries, query_lens, queries_count) }?;
325        let entries = client
326            .rt
327            .block_on(client.inner.call(List { username, queries }))?;
328        unsafe { ptr::write(out, ValetRecordIndex::from_entries(&entries)) };
329        Ok(())
330    })())
331}

333/// # Safety
334///
335/// If `count > 0`, `queries` and `query_lens` must each point to
336/// `count` initialized elements, and every non-null `queries[i]` must
337/// be valid for `query_lens[i]` bytes.
338unsafe fn collect_queries(
339    queries: *const *const c_char,
340    query_lens: *const usize,
341    count: usize,
342) -> Result<Vec<String>, FfiCallError> {
343    if count == 0 {
344        return Ok(Vec::new());
345    }
346    not_null(queries, "queries")?;
347    not_null(query_lens, "query_lens")?;
348    let ptrs = unsafe { slice::from_raw_parts(queries, count) };
349    let lens = unsafe { slice::from_raw_parts(query_lens, count) };
350    let mut out = Vec::with_capacity(count);
351    for (p, l) in ptrs.iter().zip(lens.iter()) {
352        not_null(*p, "queries[i]")?;
353        out.push(unsafe { bytes_to_string(*p, *l, "queries[i]") }?);
354    }
355    Ok(out)
356}

395/// # Safety
396///
397/// `client` must be a valid pointer previously returned by one of the
398/// `valet_ffi_client_new_*` constructors and not yet freed. `username`
399/// must point to a null-terminated UTF-8 C string. If `uuid_len > 0`,
400/// `uuid` must point to `uuid_len` initialized bytes containing the
401/// UTF-8 string representation of the UUID. `out` must be non-null
402/// and writable; on success it receives a heap-allocated
403/// [`ValetRecord`] that the caller must release with
404/// [`crate::ffi::valet_ffi_record_free`].
405#[unsafe(no_mangle)]
406pub unsafe extern "C" fn valet_ffi_client_fetch(
407    client: *mut ValetClient,
408    username: *const c_char,
409    uuid: *const c_char,
410    uuid_len: usize,
411    out: *mut ValetRecord,
412) -> i32 {
413    ffi::report((|| -> Result<(), FfiCallError> {
414        let client = unsafe { borrow(client) }?;
415        not_null(username, "username")?;
416        not_null(uuid, "uuid")?;
417        not_null(out, "out")?;
418        let username = unsafe { cstr_to_string(username, "username") }?;
419        let uuid_str = unsafe { bytes_to_string(uuid, uuid_len, "uuid") }?;
420        let uuid: Uuid<Record> =
421            Uuid::parse(&uuid_str).map_err(|_| FfiCallError::InvalidUuid("uuid"))?;
422        let record = client.rt.block_on(
423            client
424                .inner
425                .call(crate::protocol::message::Fetch { username, uuid }),
426        )?;
427        unsafe { ptr::write(out, ValetRecord::from_record(&record)) };
428        Ok(())
429    })())
430}

432/// # Safety
433///
434/// `client` must be either null or a pointer previously returned by
435/// one of the `valet_ffi_client_new_*` constructors and not yet
436/// freed. After this call the pointer must not be used again.
437#[unsafe(no_mangle)]
438pub unsafe extern "C" fn valet_ffi_client_free(client: *mut ValetClient) {
439    if !client.is_null() {
440        drop(unsafe { Box::from_raw(client) });
441    }
