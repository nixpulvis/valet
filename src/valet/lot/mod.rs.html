23/// An encrypted collection of secrets.
24///
25/// Each lot has its own _lot key_, i.e. [`Key<Lot>`] which is used to encrypt
26/// all of the records within the lot. Users with access to a lot obtain the lot
27/// key through the `user_lots` SQL table.
28///
29/// Example `user_lots` table:
30///
31/// | username | lot |    data    |   nonce    |
32/// |----------|-----|------------|------------|
33/// | Alice    | `a` | `tvuZQ1XS` | `6jLC3aP9` |
34/// | Alice    | `b` | `LyZJM8GA` | `SCW2EWjc` |
35/// | Bob      | `a` | `dWPiZfO9` | `oQ/2Y845` |
36///
37/// The lot keys they derive:
38///
39/// |  Key   | `Decrypt_A` is Alice's            | `Decrypt_B` is Bob's              |
40/// |--------|-----------------------------------|-----------------------------------|
41/// | `Ka`   | `= Decrypt_A(tvuZQ1XS, 6jLC3aP9)` | `= Decrypt_B(dWPiZfO9, oQ/2Y845)` |
42/// | `Kb`   | `= Decrypt_A(LyZJM8GA, SCW2EWjc)` | N/A                               |
43#[derive(PartialEq, Eq)]
44pub struct Lot {
45    // user: &'a User, TODO name and key are meaningless without a user.
46    uuid: Uuid<Self>,
47    name: String,
48    key: Key<Self>,
49}

82    /// Save this lot to the database, detecting and handling key rotation.
83    ///
84    /// If the lot key has changed since the last save, all records are
85    /// re-encrypted with the new key before the updated key is stored.
86    #[cfg(feature = "db")]
87    pub async fn save(&self, db: &Database, user: &User) -> Result<Uuid<Self>, Error> {
88        let uuid = self.uuid.to_string();
89        let active = self::orm::ActiveModel {
90            uuid: Unchanged(uuid.clone()),
91        };
92        self::orm::Entity::insert(active)
93            .on_conflict_do_nothing()
94            .exec(db.connection())
95            .await?;
96
97        // Load existing user_lot once to detect changes.
98        let existing_ul =
99            self::orm::user_lots::Entity::find_by_id((user.username().to_owned(), uuid.to_owned()))
100                .one(db.connection())
101                .await?;
102
103        let aad = Lot::aad(user.username(), &uuid);
104        match existing_ul {
105            None => {
106                let encrypted = user.key().encrypt_with_aad(self.key.as_bytes(), &aad)?;
107                let active = self::orm::user_lots::ActiveModel {
108                    username: Set(user.username().into()),
109                    lot_uuid: Set(uuid),
110                    name: Set(self.name.clone()),
111                    data: Set(encrypted.data),
112                    nonce: Set(encrypted.nonce),
113                };
114                self::orm::user_lots::Entity::insert(active)
115                    .exec(db.connection())
116                    .await?;
117            }
118            Some(existing) => {
119                let name_changed = existing.name != self.name;
120
121                // Detect key change by comparing the stored key with current self.key.
122                // If the current key is different, we re-encrypt all of the records in
123                // this lot under the new lot key.
124                //
125                // TODO: A mechanism for sharing the new lot key with other users will
126                // be needed, similarly to how we need a way to share a lot in the first
127                // place.
128                let existing_encrypted = Encrypted {
129                    data: existing.data.clone(),
130                    nonce: existing.nonce.clone(),
131                };
132                let existing_key_bytes = user.key().decrypt_with_aad(&existing_encrypted, &aad)?;
133                let key_changed = existing_key_bytes != self.key.as_bytes();
134
135                let mut active = existing.into_active_model();
136
137                if name_changed {
138                    active.name = Set(self.name.clone());
139                }
140
141                if key_changed {
142                    let encrypted = user.key().encrypt_with_aad(self.key.as_bytes(), &aad)?;
143                    active.data = Set(encrypted.data);
144                    active.nonce = Set(encrypted.nonce);
145
146                    self.reencrypt_records(db, &existing_key_bytes).await?;
147                }
148
149                if name_changed || key_changed {
150                    active.update(db.connection()).await?;
151                }
152            }
153        }
154
155        Ok(self.uuid.clone())
156    }

158    #[cfg(feature = "db")]
159    async fn reencrypt_records(
160        &self,
161        db: &Database,
162        existing_key_bytes: &[u8],
163    ) -> Result<(), Error> {
164        let old_key = Key::from_bytes(&existing_key_bytes);
165        let old_lot = Lot {
166            uuid: self.uuid.clone(),
167            name: self.name.clone(),
168            key: old_key,
169        };
170        let records = Record::load_all(db, &old_lot).await?;
171        for record in &records {
172            record.upsert(db, self).await?;
173        }
174        Ok(())
175    }

177    /// Load a user's lot by name.
178    #[cfg(feature = "db")]
179    pub async fn load(db: &Database, name: &str, user: &User) -> Result<Option<Self>, Error> {
180        let ul = self::orm::user_lots::Entity::find()
181            .filter(self::orm::user_lots::Column::Username.eq(user.username()))
182            .filter(self::orm::user_lots::Column::Name.eq(name))
183            .one(db.connection())
184            .await?
185            .ok_or(Error::MissingLotKey)?;
186        if let Some(model) = self::orm::Entity::find_by_id(&ul.lot_uuid)
187            .one(db.connection())
188            .await?
189        {
190            let lot = Self::decrypt_and_build(&user, model, ul)?;
191            Ok(Some(lot))
192        } else {
193            Ok(None)
194        }
195    }

197    /// Load a user's lots.
198    #[cfg(feature = "db")]
199    pub async fn load_all(db: &Database, user: &User) -> Result<Vec<Self>, Error> {
200        let uls = self::orm::user_lots::Entity::find()
201            .filter(self::orm::user_lots::Column::Username.eq(user.username()))
202            .all(db.connection())
203            .await?;
204        let mut lots = Vec::new();
205        for ul in uls {
206            if let Some(model) = self::orm::Entity::find_by_id(&ul.lot_uuid)
207                .one(db.connection())
208                .await?
209            {
210                let lot = Self::decrypt_and_build(&user, model, ul)?;
211                lots.push(lot);
212            }
213        }
214        Ok(lots)
215    }

217    /// Delete this lot, cascading to records and user_lots.

219    pub async fn delete(&self, db: &Database) -> Result<(), Error> {
220        self::orm::Entity::delete_by_id(self.uuid.to_string())
221            .exec(db.connection())
222            .await?;
223        Ok(())
224    }

226    #[cfg(feature = "db")]
227    fn decrypt_and_build(
228        user: &User,
229        model: self::orm::Model,
230        ul: self::orm::user_lots::Model,
231    ) -> Result<Lot, Error> {
232        let encrypted = Encrypted {
233            data: ul.data,
234            nonce: ul.nonce,
235        };
236        let aad = Lot::aad(user.username(), &ul.lot_uuid);
237        let key_bytes = user.key().decrypt_with_aad(&encrypted, &aad)?;
238        Ok(Lot {
239            uuid: Uuid::parse(&model.uuid)?,
240            name: ul.name,
241            key: Key::from_bytes(&key_bytes),
242        })
243    }
244
245    #[cfg(feature = "db")]
246    fn aad(username: &str, lot_uuid: &str) -> Vec<u8> {
247        [username.as_bytes(), lot_uuid.as_bytes()].concat()
248    }

455        let record = Record::show(&db, &lot, &record_uuid)

503    /// Returns the lot key for a given user/lot as decrypted from the
504    /// user_lots table.
505    async fn get_user_lot_key(db: &Database, user: &User, lot: &Lot) -> Key<Lot> {
506        let ul = self::orm::user_lots::Entity::find_by_id((
507            user.username().to_owned(),
508            lot.uuid().to_string(),
509        ))
510        .one(db.connection())
511        .await
512        .expect("failed to select user lot key")
513        .expect("missing lot key");
514        let encrypted = Encrypted {
515            data: ul.data,
516            nonce: ul.nonce,
517        };
518        let aad = Lot::aad(user.username(), &lot.uuid().to_string());
519        Key::<Lot>::from_bytes(
520            &user
521                .key()
522                .decrypt_with_aad(&encrypted, &aad)
523                .expect("failed to decrypted lot key"),
524        )
525    }
