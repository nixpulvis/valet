26/// An encrypted collection of secrets.
27///
28/// Each lot has its own _lot key_, i.e. [`Key<Lot>`] which is used to encrypt
29/// all of the records within the lot. Users with access to a lot obtain the lot
30/// key through the `user_lots` SQL table.
31///
32/// Example `user_lots` table:
33///
34/// | username | lot |    data    |   nonce    |
35/// |----------|-----|------------|------------|
36/// | Alice    | `a` | `tvuZQ1XS` | `6jLC3aP9` |
37/// | Alice    | `b` | `LyZJM8GA` | `SCW2EWjc` |
38/// | Bob      | `a` | `dWPiZfO9` | `oQ/2Y845` |
39///
40/// The lot keys they derive:
41///
42/// |  Key   | `Decrypt_A` is Alice's            | `Decrypt_B` is Bob's              |
43/// |--------|-----------------------------------|-----------------------------------|
44/// | `Ka`   | `= Decrypt_A(tvuZQ1XS, 6jLC3aP9)` | `= Decrypt_B(dWPiZfO9, oQ/2Y845)` |
45/// | `Kb`   | `= Decrypt_A(LyZJM8GA, SCW2EWjc)` | N/A                               |
46pub struct Lot {
47    uuid: Uuid<Self>,
48    name: String,
49    /// Shared so the fetcher closure installed on [`Lot::store`] can
50    /// hold an [`Arc`] clone of the same live key - no byte copy of
51    /// the secret, and one authoritative zeroize on final drop.
52    key: Arc<Key<Self>>,
53    /// Live storgit store for this lot. Opened once on
54    /// [`Lot::decrypt_and_build`] (with a fetcher that decrypts
55    /// `records.module` rows on demand under this lot's key) or
56    /// initialised fresh on [`Lot::new`]. Reused across every record
57    /// op against this lot so we pay the parent untar and gix open
58    /// costs once per session instead of once per op.
59    #[cfg(feature = "db")]
60    store: storgit::Store<SubmoduleLayout>,
61    /// Scratch dir that owns the storgit repo's on-disk location.
62    /// Kept alongside `store` so the directory lives exactly as long
63    /// as the `Lot`; dropping the lot removes the backing repo. A
64    /// future milestone will replace this with a persistent
65    /// caller-managed path so the repo survives across sessions.
66    #[cfg(feature = "db")]
67    _scratch: tempfile::TempDir,
68    /// Materialised label->uuid index for every live record in the
69    /// lot. Built from the store's label cache on construction and
70    /// kept in sync by [`Record::save`] / [`Record::delete`], which
71    /// hold `&mut Lot` for the mutation.
72    #[cfg(feature = "db")]
73    index: RecordIndex,
74}

76impl PartialEq for Lot {
77    fn eq(&self, other: &Self) -> bool {
78        // Store identity is uuid + name + key. The live storgit handle
79        // carries session-scoped state (scratch dir, dirty tracking)
80        // that is not part of the lot's persisted identity.
81        self.uuid == other.uuid && self.name == other.name && self.key == other.key
82    }
83}

165    /// Build a [`storgit::ModuleFetcher`] that resolves module bytes
166    /// for `lot_uuid` by looking them up in `records` under the given
167    /// lot key. The fetcher is sync (storgit's interface is sync), so
168    /// it bridges to the async DB via [`tokio::runtime::Handle::block_on`].
169    /// Callers must therefore invoke any storgit op that may trigger
170    /// it from inside [`tokio::task::spawn_blocking`] (or
171    /// [`tokio::task::block_in_place`]); calling from a plain async
172    /// context panics when `block_on` executes.
173    ///
174    /// The query filters on `lot_uuid`, so a record uuid claimed by a
175    /// different lot simply doesn't appear - the fetcher returns
176    /// `Ok(None)` and storgit's usual "live in parent / no backing
177    /// bytes" check surfaces the inconsistency if it matters.
178    #[cfg(feature = "db")]
179    fn make_fetcher(
180        db: Database,
181        lot_key: Arc<Key<Lot>>,
182        lot_uuid: Uuid<Lot>,
183    ) -> storgit::ModuleFetcher {
184        Arc::new(move |id: &storgit::Id| {
185            let record_uuid = Uuid::<Record>::parse(id.as_str())
186                .map_err(|e| Box::new(e) as Box<dyn std::error::Error + Send + Sync + 'static>)?;
187            let handle = tokio::runtime::Handle::current();
188            let lot_uuid_str = lot_uuid.to_string();
189            let model = handle
190                .block_on(async {
191                    record::orm::Entity::find()
192                        .filter(record::orm::Column::Uuid.eq(record_uuid.to_string()))
193                        .filter(record::orm::Column::LotUuid.eq(lot_uuid_str))
194                        .one(db.connection())
195                        .await
196                })
197                .map_err(|e| Box::new(e) as Box<dyn std::error::Error + Send + Sync + 'static>)?;
198            let Some(model) = model else {
199                return Ok(None);
200            };
201            let aad = Record::module_aad(&record_uuid, &lot_uuid);
202            let bytes = lot_key
203                .decrypt_with_aad(&Encrypted::unpack(&model.module), &aad)
204                .map_err(|e| Box::new(e) as Box<dyn std::error::Error + Send + Sync + 'static>)?;
205            Ok(Some(bytes))
206        })
207    }

231    /// Persist this lot and its binding to `user`.
232    ///
233    /// Upserts the `lots` row (uuid + encrypted parent tarball) when the
234    /// live store has a dirty parent to flush, then writes or updates the
235    /// per-user `user_lots` row binding `user` to this lot under the
236    /// user's key. Only the lot name is mutable on an existing
237    /// `user_lots` row; lot-key rotation is not supported.
238    #[cfg(feature = "db")]
239    pub async fn save(&mut self, db: &Database, user: &User) -> Result<Uuid<Self>, Error> {
240        let uuid = self.uuid.to_string();
241        // Persist whatever parent state the store currently has. A
242        // fresh store snapshots an empty-parent tarball (dirty on
243        // open); a loaded store with no mutations returns None and
244        // we skip the write. We upsert on conflict so a dirty parent
245        // flushed through here overwrites the existing row rather
246        // than being discarded.
247        if let Some(parent_bytes) = self
248            .store
249            .snapshot()
250            .map_err(|e| Error::Record(record::Error::Storgit(e)))?
251            .parent
252        {
253            let initial_store = self.encrypt_store(&parent_bytes)?;
254            let active = self::orm::ActiveModel {
255                uuid: Unchanged(uuid.clone()),
256                store: Set(initial_store),
257            };
258            let on_conflict = sea_orm::sea_query::OnConflict::column(self::orm::Column::Uuid)
259                .update_column(self::orm::Column::Store)
260                .to_owned();
261            self::orm::Entity::insert(active)
262                .on_conflict(on_conflict)
263                .exec(db.connection())
264                .await?;
265        }
266
267        // Load existing user_lot once to detect changes.
268        let existing_ul =
269            self::orm::user_lots::Entity::find_by_id((user.username().to_owned(), uuid.to_owned()))
270                .one(db.connection())
271                .await?;
272
273        let aad = Lot::user_lot_aad(user.username(), &self.uuid);
274        match existing_ul {
275            None => {
276                let encrypted = user.key().encrypt_with_aad(self.key.as_bytes(), &aad)?;
277                let active = self::orm::user_lots::ActiveModel {
278                    username: Set(user.username().into()),
279                    lot_uuid: Set(uuid),
280                    name: Set(self.name.clone()),
281                    data: Set(encrypted.data),
282                    nonce: Set(encrypted.nonce),
283                };
284                self::orm::user_lots::Entity::insert(active)
285                    .exec(db.connection())
286                    .await?;
287            }
288            Some(existing) => {
289                // Only name changes are supported for existing rows.
290                //
291                // TODO: lot-key rotation. Needs to re-encrypt the
292                // parent tarball and every records.module row under
293                // the new key, coordinate that with any other users
294                // sharing the lot, and survive a crash mid-rotation.
295                // The previous implementation did none of those
296                // well, so it was removed rather than left as a
297                // footgun.
298                if existing.name != self.name {
299                    let mut active = existing.into_active_model();
300                    active.name = Set(self.name.clone());
301                    active.update(db.connection()).await?;
302                }
303            }
304        }
305
306        Ok(self.uuid.clone())
307    }

309    /// Load a user's lot by name.
310    #[cfg(feature = "db")]
311    pub async fn load(db: &Database, name: &str, user: &User) -> Result<Option<Self>, Error> {
312        let ul = self::orm::user_lots::Entity::find()
313            .filter(self::orm::user_lots::Column::Username.eq(user.username()))
314            .filter(self::orm::user_lots::Column::Name.eq(name))
315            .one(db.connection())
316            .await?
317            .ok_or(Error::MissingLotKey)?;
318        if let Some(model) = self::orm::Entity::find_by_id(&ul.lot_uuid)
319            .one(db.connection())
320            .await?
321        {
322            let lot = Self::decrypt_and_build(db, user, model, ul)?;
323            Ok(Some(lot))
324        } else {
325            Ok(None)
326        }
327    }

329    /// Load a user's lots.
330    #[cfg(feature = "db")]
331    pub async fn load_all(db: &Database, user: &User) -> Result<Vec<Self>, Error> {
332        let uls = self::orm::user_lots::Entity::find()
333            .filter(self::orm::user_lots::Column::Username.eq(user.username()))
334            .all(db.connection())
335            .await?;
336        let mut lots = Vec::new();
337        for ul in uls {
338            if let Some(model) = self::orm::Entity::find_by_id(&ul.lot_uuid)
339                .one(db.connection())
340                .await?
341            {
342                let lot = Self::decrypt_and_build(db, user, model, ul)?;
343                lots.push(lot);
344            }
345        }
346        Ok(lots)
347    }

361    #[cfg(feature = "db")]
362    fn decrypt_and_build(
363        db: &Database,
364        user: &User,
365        model: self::orm::Model,
366        ul: self::orm::user_lots::Model,
367    ) -> Result<Lot, Error> {
368        let uuid = Uuid::<Lot>::parse(&model.uuid)?;
369        let encrypted = Encrypted {
370            data: ul.data,
371            nonce: ul.nonce,
372        };
373        let aad = Lot::user_lot_aad(user.username(), &uuid);
374        let key_bytes = user.key().decrypt_with_aad(&encrypted, &aad)?;
375        let key = Arc::new(Key::<Lot>::from_bytes(&key_bytes));
376
377        // Decrypt the parent tarball under the (just-derived) lot key.
378        let store_aad = Lot::store_aad(&uuid);
379        let parent_bytes = key.decrypt_with_aad(&Encrypted::unpack(&model.store), &store_aad)?;
380
381        let fetcher = Lot::make_fetcher(db.clone(), key.clone(), uuid.clone());
382        let scratch = tempfile::Builder::new()
383            .prefix("valet-lot-")
384            .tempdir()
385            .map_err(|e| Error::Record(record::Error::Storgit(storgit::Error::Io(e))))?;
386        let store = storgit::Store::<SubmoduleLayout>::new(scratch.path().join("repo"))
387            .and_then(|s| {
388                s.with_parts(storgit::Parts {
389                    parent: parent_bytes,
390                    modules: storgit::Modules::new(),
391                    fetcher: Some(fetcher),
392                })
393            })
394            .map_err(|e| Error::Record(record::Error::Storgit(e)))?;
395        let index = RecordIndex::from_store(&store).map_err(Error::Record)?;
396
397        Ok(Lot {
398            uuid,
399            name: ul.name,
400            key,
401            store,
402            _scratch: scratch,
403            index,
404        })
405    }

407    /// Encrypt the storgit parent tarball under this lot's key with the
408    /// lot-scoped store AAD. Packed for the `lots.store` column.
409    #[cfg(feature = "db")]
410    pub(crate) fn encrypt_store(&self, parent_bytes: &[u8]) -> Result<Vec<u8>, encrypt::Error> {
411        let aad = Lot::store_aad(&self.uuid);
412        let encrypted = self.key.encrypt_with_aad(parent_bytes, &aad)?;
413        Ok(encrypted.pack())
414    }

613        let user_lot = self::orm::user_lots::Entity::find_by_id((user.username().to_owned(), uuid))
614            .one(db.connection())
615            .await
616            .expect("failed to load user_lot");
617        assert!(user_lot.is_none());
618    }

620    /// Returns the lot key for a given user/lot as decrypted from the
621    /// user_lots table.
622    #[cfg(feature = "db")]
623    async fn get_user_lot_key(db: &Database, user: &User, lot: &Lot) -> Key<Lot> {
624        let ul = self::orm::user_lots::Entity::find_by_id((
625            user.username().to_owned(),
626            lot.uuid().to_string(),
627        ))
628        .one(db.connection())
629        .await
630        .expect("failed to select user lot key")
631        .expect("missing lot key");
632        let encrypted = Encrypted {
633            data: ul.data,
634            nonce: ul.nonce,
635        };
636        let aad = Lot::user_lot_aad(user.username(), lot.uuid());
637        Key::<Lot>::from_bytes(
638            &user
639                .key()
640                .decrypt_with_aad(&encrypted, &aad)
641                .expect("failed to decrypted lot key"),
642        )
643    }
