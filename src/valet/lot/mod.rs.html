17/// An encrypted collection of secrets.
18///
19/// Each lot has its own _lot key_, i.e. [`Key<Lot>`] which is used to encrypt
20/// all of the records within the lot. Users with access to a lot obtain the lot
21/// key through the `user_lots` SQL table.
22///
23/// Example `user_lots` table:
24///
25/// | username | lot |    data    |   nonce    |
26/// |----------|-----|------------|------------|
27/// | Alice    | `a` | `tvuZQ1XS` | `6jLC3aP9` |
28/// | Alice    | `b` | `LyZJM8GA` | `SCW2EWjc` |
29/// | Bob      | `a` | `dWPiZfO9` | `oQ/2Y845` |
30///
31/// The lot keys they derive:
32///
33/// |  Key   | `Decrypt_A` is Alice's            | `Decrypt_B` is Bob's              |
34/// |--------|-----------------------------------|-----------------------------------|
35/// | `Ka`   | `= Decrypt_A(tvuZQ1XS, 6jLC3aP9)` | `= Decrypt_B(dWPiZfO9, oQ/2Y845)` |
36/// | `Kb`   | `= Decrypt_A(LyZJM8GA, SCW2EWjc)` | N/A                               |
37#[derive(PartialEq, Eq)]
38pub struct Lot {
39    // user: &'a User, TODO name and key are meaningless without a user.
40    uuid: Uuid<Self>,
41    name: String,
42    key: Key<Self>,
43}

71    /// Save this lot to the database, detecting and handling key rotation.
72    ///
73    /// If the lot key has changed since the last save, all records are
74    /// re-encrypted with the new key before the updated key is stored.
75    pub async fn save(&self, db: &Database, user: &User) -> Result<Uuid<Self>, Error> {
76        let uuid = self.uuid.to_string();
77        let active = self::orm::ActiveModel {
78            uuid: Unchanged(uuid.clone()),
79        };
80        self::orm::Entity::insert(active)
81            .on_conflict_do_nothing()
82            .exec(db.connection())
83            .await?;
84
85        // Load existing user_lot once to detect changes.
86        let existing_ul =
87            self::orm::user_lots::Entity::find_by_id((user.username().to_owned(), uuid.to_owned()))
88                .one(db.connection())
89                .await?;
90
91        let aad = Lot::aad(user.username(), &uuid);
92        match existing_ul {
93            None => {
94                let encrypted = user.key().encrypt_with_aad(self.key.as_bytes(), &aad)?;
95                let active = self::orm::user_lots::ActiveModel {
96                    username: Set(user.username().into()),
97                    lot_uuid: Set(uuid),
98                    name: Set(self.name.clone()),
99                    data: Set(encrypted.data),
100                    nonce: Set(encrypted.nonce),
101                };
102                self::orm::user_lots::Entity::insert(active)
103                    .exec(db.connection())
104                    .await?;
105            }
106            Some(existing) => {
107                let name_changed = existing.name != self.name;
108
109                // Detect key change by comparing the stored key with current self.key.
110                // If the current key is different, we re-encrypt all of the records in
111                // this lot under the new lot key.
112                //
113                // TODO: A mechanism for sharing the new lot key with other users will
114                // be needed, similarly to how we need a way to share a lot in the first
115                // place.
116                let existing_encrypted = Encrypted {
117                    data: existing.data.clone(),
118                    nonce: existing.nonce.clone(),
119                };
120                let existing_key_bytes = user.key().decrypt_with_aad(&existing_encrypted, &aad)?;
121                let key_changed = existing_key_bytes != self.key.as_bytes();
122
123                let mut active = existing.into_active_model();
124
125                if name_changed {
126                    active.name = Set(self.name.clone());
127                }
128
129                if key_changed {
130                    let encrypted = user.key().encrypt_with_aad(self.key.as_bytes(), &aad)?;
131                    active.data = Set(encrypted.data);
132                    active.nonce = Set(encrypted.nonce);
133
134                    self.reencrypt_records(db, &existing_key_bytes).await?;
135                }
136
137                if name_changed || key_changed {
138                    active.update(db.connection()).await?;
139                }
140            }
141        }
142
143        Ok(self.uuid.clone())
144    }
145
146    async fn reencrypt_records(
147        &self,
148        db: &Database,
149        existing_key_bytes: &[u8],
150    ) -> Result<(), Error> {
151        let old_key = Key::from_bytes(&existing_key_bytes);
152        let old_lot = Lot {
153            uuid: self.uuid.clone(),
154            name: self.name.clone(),
155            key: old_key,
156        };
157        let records = Record::load_all(db, &old_lot).await?;
158        for record in &records {
159            record.upsert(db, self).await?;
160        }
161        Ok(())
162    }

164    /// Load a user's lot by name.
165    pub async fn load(db: &Database, name: &str, user: &User) -> Result<Option<Self>, Error> {
166        let ul = self::orm::user_lots::Entity::find()
167            .filter(self::orm::user_lots::Column::Username.eq(user.username()))
168            .filter(self::orm::user_lots::Column::Name.eq(name))
169            .one(db.connection())
170            .await?
171            .ok_or(Error::MissingLotKey)?;
172        if let Some(model) = self::orm::Entity::find_by_id(&ul.lot_uuid)
173            .one(db.connection())
174            .await?
175        {
176            let lot = Self::decrypt_and_build(&user, model, ul)?;
177            Ok(Some(lot))
178        } else {
179            Ok(None)
180        }
181    }
182
183    /// Load a user's lots.
184    pub async fn load_all(db: &Database, user: &User) -> Result<Vec<Self>, Error> {
185        let uls = self::orm::user_lots::Entity::find()
186            .filter(self::orm::user_lots::Column::Username.eq(user.username()))
187            .all(db.connection())
188            .await?;
189        let mut lots = Vec::new();
190        for ul in uls {
191            if let Some(model) = self::orm::Entity::find_by_id(&ul.lot_uuid)
192                .one(db.connection())
193                .await?
194            {
195                let lot = Self::decrypt_and_build(&user, model, ul)?;
196                lots.push(lot);
197            }
198        }
199        Ok(lots)
200    }
201
202    /// Delete this lot, cascading to records and user_lots.
203    pub async fn delete(&self, db: &Database) -> Result<(), Error> {
204        self::orm::Entity::delete_by_id(self.uuid.to_string())
205            .exec(db.connection())
206            .await?;
207        Ok(())
208    }
209
210    fn decrypt_and_build(
211        user: &User,
212        model: self::orm::Model,
213        ul: self::orm::user_lots::Model,
214    ) -> Result<Lot, Error> {
215        let encrypted = Encrypted {
216            data: ul.data,
217            nonce: ul.nonce,
218        };
219        let aad = Lot::aad(user.username(), &ul.lot_uuid);
220        let key_bytes = user.key().decrypt_with_aad(&encrypted, &aad)?;
221        Ok(Lot {
222            uuid: Uuid::parse(&model.uuid)?,
223            name: ul.name,
224            key: Key::from_bytes(&key_bytes),
225        })
226    }
227
228    fn aad(username: &str, lot_uuid: &str) -> Vec<u8> {
229        [username.as_bytes(), lot_uuid.as_bytes()].concat()
230    }

242#[derive(Debug)]
243pub enum Error {
244    MissingLotKey,
245    Uuid(crate::uuid::Error),
246    Encrypt(encrypt::Error),
247    Record(record::Error),
248    Database(db::Error),
249}
250
251impl From<crate::uuid::Error> for Error {
252    fn from(err: crate::uuid::Error) -> Self {
253        Error::Uuid(err)
254    }
255}
256
257impl From<encrypt::Error> for Error {
258    fn from(err: encrypt::Error) -> Self {
259        Error::Encrypt(err)
260    }
261}
262
263impl From<db::Error> for Error {
264    fn from(err: db::Error) -> Self {
265        Error::Database(err)
266    }
267}
268
269impl From<sea_orm::DbErr> for Error {
270    fn from(err: sea_orm::DbErr) -> Self {
271        Error::Database(err.into())
272    }
273}
274
275impl From<record::Error> for Error {
276    fn from(err: record::Error) -> Self {
277        Error::Record(err)
278    }

441    /// Returns the lot key for a given user/lot as decrypted from the
442    /// user_lots table.
443    async fn get_user_lot_key(db: &Database, user: &User, lot: &Lot) -> Key<Lot> {
444        let ul = self::orm::user_lots::Entity::find_by_id((
445            user.username().to_owned(),
446            lot.uuid().to_string(),
447        ))
448        .one(db.connection())
449        .await
450        .expect("failed to select user lot key")
451        .expect("missing lot key");
452        let encrypted = Encrypted {
453            data: ul.data,
454            nonce: ul.nonce,
455        };
456        let aad = Lot::aad(user.username(), &lot.uuid().to_string());
457        Key::<Lot>::from_bytes(
458            &user
459                .key()
460                .decrypt_with_aad(&encrypted, &aad)
461                .expect("failed to decrypted lot key"),
462        )
463    }
