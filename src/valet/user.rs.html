10/// A user of valet, who is uniquely identified by username.
11///
12/// As is standard practice with password handling, the user's password provided
13/// to either [`User::new`] or [`User::load`] is never saved anywhere and is
14/// kept in memory for as little time as possible.
15///
16/// The user's password (and a random saved "salt") is used to derive the _user
17/// key_, i.e. [`Key<User>`]. To generate this key we use a common Key
18/// Derivation Function (KDF), namely [`argon2`][argon2]. Each user record saves
19/// it's random salt value in order to prevent users with the same password from
20/// getting the same key, and thus opening up the scheme to ["rainbow table"][1]
21/// attacks.
22///
23/// In addition to the salt, each user also stores a short encrypted validation
24/// string which is used to authenticate the user. Simply being able
25/// to decrpyt the string is enough to verify the user, since we use
26/// ["Authenticated Encryption"][2] (the AE in AEAD).
27///
28/// [1]: https://en.wikipedia.org/wiki/Rainbow_table
29/// [2]: https://en.wikipedia.org/wiki/Authenticated_encryption
30/// [argon2]: https://docs.rs/argon2/latest/argon2
31#[derive(PartialEq, Eq)]
32pub struct User {
33    username: String,
34    salt: [u8; SALT_SIZE],
35    validation: Encrypted,
36    key: Key<Self>,
37}
38
39impl User {
40    pub fn new(username: &str, password: Password) -> Result<Self, Error> {
41        let salt = encrypt::generate_salt();
42        let key = Key::from_password(password, &salt)?;
43        let validation = key.encrypt(VALIDATION)?;
44        Ok(User {
45            username: username.into(),
46            salt,
47            validation,
48            key,
49        })
50    }
51
52    pub fn username(&self) -> &str {
53        &self.username
54    }
55
56    pub fn key(&self) -> &Key<Self> {
57        &self.key
58    }
59
60    pub fn validate(&self) -> bool {
61        if let Ok(v) = self.key().decrypt(&self.validation) {
62            v == VALIDATION // This should never be false.
63        } else {
64            false
65        }
66    }
67
68    // TODO: Return type, insert or update info.
69    pub async fn register(self, db: &Database) -> Result<Self, Error> {
70        let sql_user = db::users::SqlUser {
71            username: self.username.clone(),
72            salt: self.salt.to_vec(),
73            validation_data: self.validation.data.clone(),
74            validation_nonce: self.validation.nonce.clone(),
75        };
76        sql_user.insert(&db).await?;
77        Ok(self)
78    }
79
80    pub async fn load<'a>(
81        db: &'a Database,
82        username: &'a str,
83        password: Password<'a>,
84    ) -> Result<Self, Error> {
85        let sql_user = db::users::SqlUser::select(&db, &username).await?;
86        let key = Key::from_password(password, &sql_user.salt[..])?;
87        let validation = Encrypted {
88            data: sql_user.validation_data,
89            nonce: sql_user.validation_nonce,
90        };
91        let user = User {
92            username: sql_user.username,
93            salt: sql_user.salt.try_into().map_err(|_| Error::SaltError)?,
94            validation,
95            key,
96        };
97        if user.validate() {
98            Ok(user)
99        } else {
100            Err(Error::Invalid)
101        }
102    }
103
104    /// Load all of this user's lots.
105    ///
106    /// This function as well as [`Lot::load`] and [`Lot::load_all`] utilize the
107    /// `user_lots` SQL table to determine lot membership as well as to access
108    /// the user encrypted lot key for each lot.
109    ///
110    /// For more information, see [`Lot`].
111    pub async fn lots(&self, db: &Database) -> Result<Vec<Lot>, Error> {
112        Ok(Lot::load_all(&db, self).await?)
113    }
114}

124#[derive(Debug)]
125pub enum Error {
126    Invalid,
127    SaltError,
128    Encrypt(encrypt::Error),
129    Database(db::Error),
130    Lot(lot::Error),
131}
132
133impl From<encrypt::Error> for Error {
134    fn from(err: encrypt::Error) -> Self {
135        Error::Encrypt(err)
136    }
137}
138
139impl From<db::Error> for Error {
140    fn from(err: db::Error) -> Self {
141        Error::Database(err)
142    }
143}

145impl From<lot::Error> for Error {

163    #[test]
164    fn invalid() {
165        let mut user = User::new("alice", pw!("password")).expect("failed to create user");
166        let imposter = User::new("charlie", pw!("password")).expect("failed to create user");
167        user.validation = imposter
168            .key()
169            .encrypt(VALIDATION)
170            .expect("failed to encrypt");
171        assert!(!user.validate());
172    }
