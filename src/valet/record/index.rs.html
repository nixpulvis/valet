16    label_nonce: Vec<u8>,

19/// An in-memory map from `Label` to `Uuid<Record>` for a single lot.
20///
21/// Built by decrypting only the `label` column of each row in the lot, which
22/// avoids materializing any passwords. Use [`RecordIndex::find`] to resolve a
23/// label to a UUID, then call [`Record::show`](crate::record::Record::show) to
24/// pull the password for that one record.
25///
26/// The index is not persisted; it's rebuilt from the encrypted label column
27/// on every [`RecordIndex::load`]. This keeps the DB as the single source of
28/// truth and removes any sync burden on `upsert`/`delete`.
29pub struct RecordIndex {
30    entries: BTreeMap<Label, Uuid<Record>>,
31}
32
33impl RecordIndex {
34    /// Build an index for `lot` by fetching and decrypting every record's
35    /// label column.
36    ///
37    /// The password-bearing `data` column isn't loaded.
38    pub async fn load(db: &Database, lot: &Lot) -> Result<Self, Error> {
39        let rows = super::orm::Entity::find()
40            .filter(super::orm::Column::LotUuid.eq(lot.uuid().to_string()))
41            .into_partial_model::<LabelRow>()
42            .all(db.connection())
43            .await?;
44
45        let mut entries = BTreeMap::new();
46        for row in rows {
47            let uuid = Uuid::<Record>::parse(&row.uuid)?;
48            let aad = Record::label_aad(&uuid, lot.uuid());
49            let encrypted = Encrypted {
50                data: row.label,
51                nonce: row.label_nonce,
52            };
53            let label = Label::decrypt_with_aad(&encrypted, lot.key(), &aad)?;
54            entries.insert(label, uuid);
55        }
56        Ok(RecordIndex { entries })
57    }
58
59    /// Look up the UUID of the record with the given label, if one exists.
60    pub fn find(&self, label: &Label) -> Option<&Uuid<Record>> {
61        self.entries.get(label)
62    }
63
64    /// Return an iterator over every label in the index.
65    ///
66    /// Iteration follows `Label: Ord`.
67    pub fn labels(&self) -> impl Iterator<Item = &Label> {
68        self.entries.keys()
69    }
70
71    /// Return an iterator over every `(label, uuid)` pair in the index.
72    ///
73    /// Iteration follows `Label: Ord`.
74    ///
75    /// Also available via `IntoIterator` for `&RecordIndex`, so `for entry in
76    /// &index` works directly.
77    pub fn iter(&self) -> impl Iterator<Item = (&Label, &Uuid<Record>)> {
78        self.into_iter()
79    }
80
81    /// Return every `(label, uuid)` pair whose label satisfies `query`.
82    ///
83    /// O(n) scan over the in-memory index. Entries yield in `Label: Ord`
84    /// order, which ties-breaks on `extra` — so labels that differ only in
85    /// their `extra` contents have an unspecified relative order. Don't
86    /// depend on it.
87    pub fn search<'a>(
88        &'a self,
89        query: &'a Query,
90    ) -> impl Iterator<Item = (&'a Label, &'a Uuid<Record>)> {
91        self.entries
92            .iter()
93            .filter(move |(label, _)| query.matches_label(label))
94    }
95
96    /// True if the lot has no records.
97    pub fn is_empty(&self) -> bool {
98        self.entries.is_empty()
99    }
100
101    /// Number of records in the lot.
102    pub fn len(&self) -> usize {
103        self.entries.len()
104    }
105}
106
107impl<'a> IntoIterator for &'a RecordIndex {
108    type Item = (&'a Label, &'a Uuid<Record>);
109    type IntoIter = std::collections::btree_map::Iter<'a, Label, Uuid<Record>>;
110
111    fn into_iter(self) -> Self::IntoIter {
112        self.entries.iter()
113    }
114}

116impl IntoIterator for RecordIndex {
117    type Item = (Label, Uuid<Record>);
118    type IntoIter = std::collections::btree_map::IntoIter<Label, Uuid<Record>>;
119
120    fn into_iter(self) -> Self::IntoIter {
121        self.entries.into_iter()
122    }

312    fn search_labels(index: &RecordIndex, q: &Query) -> Vec<String> {
313        let mut out: Vec<String> = index.search(q).map(|(l, _)| l.name().to_string()).collect();
314        out.sort();
315        out
316    }
