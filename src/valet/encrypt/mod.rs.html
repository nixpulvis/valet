1use rand_core::{OsRng, RngCore};

3// This value can be anything really, but is generally recommended to be about
4// 128-bits. The idea is that it just needs to contain more entropy than the
5// user's password.
6pub(crate) const SALT_SIZE: usize = 128 / 8;
7
8pub(crate) fn generate_salt() -> [u8; SALT_SIZE] {
9    let mut salt = [0; SALT_SIZE];
10    OsRng.fill_bytes(&mut salt);
11    salt
12}
