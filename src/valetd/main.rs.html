1//! `valetd` - the Valet daemon.
2//!
3//! One process, two I/O loops selected by
4//! `--transport=socket|native-messaging`. Auto-detect when the flag is
5//! absent: stdin is a terminal -> `socket`; stdin is a pipe ->
6//! `native-messaging` (the way browsers launch native-host binaries).
7//!
8//! Socket transport: binds `$VALET_SOCKET` (or the default path) and
9//! serves a `Client<Embedded>` owning a SQLite handle.
10//!
11//! Native-messaging transport: reads/writes the browser's stdio
12//! native-messaging envelope. `VALET_BACKEND=auto|socket|embedded`
13//! picks what handler fields the requests:
14//!
15//! * `embedded` - a `Client<Embedded>` owning its own SQLite handle.
16//! * `socket` - a `Client<Socket>` that relays to another long-lived
17//!   `valetd --transport=socket`.
18//! * `auto` (default) - probe the socket; fall back to embedded.
19//!
20//! Each transport and backend is gated on the corresponding
21//! `protocol-*` feature. A build that leaves one out still produces a
22//! `valetd` binary; the missing paths are simply rejected at runtime
23//! with a feature-specific error.
24

60/// Valet daemon.
61///
62/// Without `--transport`, auto-detects: stdin is a terminal -> socket;
63/// stdin is a pipe (a browser launched us) -> native-messaging.
64#[derive(Debug, Parser)]
65#[command(
66    name = "valetd",
67    about = "Valet daemon.",
68    after_help = "Environment:\n  \
69        VALET_SOCKET   Socket path for --transport=socket and for the\n                 \
70                       native-messaging socket-relay backend.\n  \
71        VALET_DB       SQLite URL for the embedded backend.\n  \
72        VALET_BACKEND  For --transport=native-messaging only:\n                 \
73                       auto|socket|embedded (default: auto)."
74)]
75struct Args {
76    /// Transport to serve. Auto-detected from stdin when omitted.
77    #[arg(long, value_enum)]
78    transport: Option<Transport>,
79
80    /// Browsers launch native-messaging hosts with extra positional
81    /// argv (Firefox: extension id, manifest path; Chrome: caller
82    /// origin). Swallow and ignore them.
83    #[arg(trailing_var_arg = true, hide = true)]
84    extras: Vec<String>,
85}
86
87fn auto_transport() -> Transport {
88    // The browser launches native-host binaries with piped stdio, so
89    // a non-terminal stdin is the strongest indicator we should speak
90    // native-messaging. A user launching from a shell gets the socket
91    // listener. systemd/launchd deployments should pass
92    // `--transport=socket` explicitly.
93    if std::io::stdin().is_terminal() {
94        Transport::Socket
95    } else {
96        Transport::NativeMessaging
97    }
98}

100pub(crate) fn parse_backend() -> Result<Backend, String> {
101    match std::env::var("VALET_BACKEND").as_deref() {
102        Ok("") | Ok("auto") | Err(_) => Ok(Backend::Auto),
103        Ok("socket") => Ok(Backend::Socket),
104        Ok("embedded") => Ok(Backend::Embedded),
105        Ok(other) => Err(format!(
106            "VALET_BACKEND={other}: expected 'auto', 'socket', or 'embedded'"
107        )),
108    }
109}

111#[tokio::main(flavor = "multi_thread")]
112async fn main() {
113    valet::logging::init();
114    let args = Args::parse();
115    for extra in &args.extras {
116        tracing::debug!(arg = %extra, "ignoring positional argument");
117    }
118    let transport = args.transport.unwrap_or_else(auto_transport);
119
120    let result = match transport {
121        Transport::Socket => run_socket().await,
122        Transport::NativeMessaging => {
123            let backend = match parse_backend() {
124                Ok(b) => b,
125                Err(err) => {
126                    error!("{err}");
127                    std::process::exit(1);
128                }
129            };
130            run_native_msg(backend).await
131        }
132    };
133    if let Err(err) = result {
134        error!("{err}");
135        std::process::exit(1);
136    }
137}

139#[cfg(all(feature = "protocol-socket", feature = "protocol-embedded"))]
140async fn run_socket() -> Result<(), String> {
141    socket_cli::run().await
142}
143
144#[cfg(not(all(feature = "protocol-socket", feature = "protocol-embedded")))]
145async fn run_socket() -> Result<(), String> {
146    Err("socket transport is disabled in this build \
147         (requires features `protocol-socket` and `protocol-embedded`)"
148        .to_string())
149}
150
151#[cfg(feature = "protocol-native-msg-server")]
152async fn run_native_msg(backend: Backend) -> Result<(), String> {
153    native_msg_cli::run(backend).await
154}
155
156#[cfg(not(feature = "protocol-native-msg-server"))]
157async fn run_native_msg(_backend: Backend) -> Result<(), String> {
158    Err("native-messaging transport is disabled in this build \
159         (requires feature `protocol-native-msg-server`)"
160        .to_string())
161}
