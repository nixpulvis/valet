28/// Parent bare repo directory inside the scratch dir.
29const PARENT_DIR: &str = "parent.git";
30/// Directory holding per-entry submodule bare repos inside the scratch
31/// dir. Each `<id>.git/` is a full bare repo with its own objects.
32const MODULES_DIR: &str = "modules";

34/// Map from entry [`EntryId`] to the bytes of that module's tarball.
35pub type Modules = HashMap<EntryId, Vec<u8>>;
36
37/// Caller-supplied lookup for module bytes. Called by
38/// [`SubmoduleLayout`] when an operation first touches an id whose
39/// bytes are neither on disk nor in [`Parts::modules`]. A fetcher is
40/// assumed to reflect live backing storage at call time. Install one
41/// with [`Store::<SubmoduleLayout>::with_fetcher`].
42///
43/// Return values:
44/// - `Ok(Some(bytes))` - module exists, here are its tarball bytes.
45/// - `Ok(None)` - no such module in backing storage. If the id is
46///   live in the parent's gitlink set, this surfaces as an error
47///   (the caller's backing store has diverged from the parent);
48///   otherwise the id is treated as fresh.
49/// - `Err(e)` - lookup itself failed; the op fails with
50///   [`crate::Error::Fetch`].
51pub type ModuleFetcher = Arc<
52    dyn Fn(&EntryId) -> Result<Option<Vec<u8>>, Box<dyn std::error::Error + Send + Sync + 'static>>
53        + Send
54        + Sync,
55>;

57/// The persisted form of a submodule-layout [`Store`].
58///
59/// Feed this to [`Store::<SubmoduleLayout>::with_parts`] to seed a
60/// store with previously-persisted bytes. An empty `parent` means
61/// "fresh store"; the first [`Store::<SubmoduleLayout>::snapshot`]
62/// will emit the newly-initialised state so the caller can persist
63/// it.
64///
65/// Whether the store is lazy is an orthogonal concern, controlled by
66/// [`Store::<SubmoduleLayout>::with_fetcher`]:
67///
68/// - No fetcher: the caller promises [`Parts::modules`] lists every
69///   live module. A miss for a live id at `ensure_loaded` time is an
70///   error; a miss for an unknown id is treated as fresh.
71/// - With a fetcher: [`Parts::modules`] is a prewarm cache consulted
72///   first; misses fall through to the fetcher, whose answer is
73///   authoritative.
74#[derive(Debug, Default)]
75pub struct Parts {
76    /// Tarball of the parent bare repo.
77    pub parent: Vec<u8>,
78    /// Tarball of each submodule bare repo the caller has in hand,
79    /// keyed by entry [`EntryId`]. With no fetcher, this is the complete
80    /// set of live modules; with a fetcher, it's an optional prewarm
81    /// cache consulted before the fetcher.
82    pub modules: Modules,
83}
84
85impl Parts {
86    /// Fold a [`Snapshot`] delta into this [`Parts`]. After applying,
87    /// `self` reflects the store's state at the moment the snapshot was
88    /// taken, and can be fed straight back into
89    /// [`Store::<SubmoduleLayout>::open`].
90    ///
91    /// - A `Some` parent in the snapshot overwrites [`Parts::parent`].
92    /// - [`ModuleChange::Changed`] overwrites or inserts the module.
93    /// - [`ModuleChange::Deleted`] removes the module if present.
94    pub fn apply(&mut self, snap: Snapshot) {
95        if let Some(parent) = snap.parent {
96            self.parent = parent;
97        }
98        for (id, change) in snap.modules {
99            match change {
100                ModuleChange::Changed(bytes) => {
101                    self.modules.insert(id, bytes);
102                }
103                ModuleChange::Deleted => {
104                    self.modules.remove(&id);
105                }
106            }
107        }
108    }
109}

111impl From<Snapshot> for Parts {
112    /// Build a fresh [`Parts`] by calling [`Parts::apply`] on an
113    /// empty one with `snap`.
114    ///
115    /// Use this only when the snapshot is known to describe the entire store,
116    /// not a delta against something already persisted. In practice that means
117    /// the first snapshot taken from a store opened with [`Parts::default`]:
118    /// every module is dirty, the parent is dirty, and there are no deletions
119    /// to express because there was nothing there to delete.
120    ///
121    /// For any subsequent snapshot, apply it to the [`Parts`] you already have.
122    /// A plain conversion there would drop every module not mentioned in the
123    /// delta, since each snapshot only contains the changes from the last.
124    fn from(snap: Snapshot) -> Self {
125        let mut parts = Parts::default();
126        parts.apply(snap);
127        parts
128    }
129}

131/// The delta produced by [`Store::<SubmoduleLayout>::snapshot`]: only
132/// the parts touched since the previous snapshot (or, for the first
133/// call, since [`Store::<SubmoduleLayout>::open`]).
134#[derive(Debug, Default)]
135pub struct Snapshot {
136    /// `Some` when the parent tarball changed and should be repersisted.
137    pub parent: Option<Vec<u8>>,
138    /// Touched submodules only. Ids absent from this map are
139    /// unchanged in storage.
140    pub modules: HashMap<EntryId, ModuleChange>,
141}

143/// What happened to a submodule between two snapshots.
144#[derive(Debug, Clone)]
145pub enum ModuleChange {
146    /// The submodule's tarball; write it to storage.
147    Changed(Vec<u8>),
148    /// The submodule was deleted; drop it from storage.
149    Deleted,
150}

152#[derive(Debug, Clone, Copy, PartialEq, Eq)]
153enum ModuleDirt {
154    Changed,
155    Deleted,

158/// Backed by a directory at `path`, containing `parent.git/` (a bare parent
159/// repo) and `modules/<id>.git/` (one bare submodule per entry).
160///
161/// # Disk layout
162///
163/// ```text
164/// <path>/
165///   parent.git/           bare repo: the store's index
166///     HEAD -> refs/heads/storgit
167///     refs/heads/storgit  -> the single parent commit
168///   modules/
169///     <id>.git/           bare repo: one per live entry, its own object DB
170/// ```
171///
172/// # Parent tree
173///
174/// The parent repo's `refs/heads/storgit` points at a single commit whose
175/// tree holds:
176///
177/// - one gitlink per live entry, filename is the entry's [`EntryId`], oid is the
178///   submodule's head commit
179/// - `.gitmodules`, a serialised manifest of the gitlink set, so the parent
180///   is a valid git submodule parent that `git submodule` can drive
181/// - `index/`, a subtree of blobs keyed by [`EntryId`] carrying each entry's
182///   label bytes. This mirrors the label cache without having to clone the
183///   entry's submodule to read it
184///
185/// A _gitlink_ is a tree entry whose blob is actually a commit ID in another
186/// repository. Git treats it as a pointer to that commit rather than as file
187/// content, which is how submodules pin a specific revision of a nested repo
188/// from their parent's tree.
189///
190/// The parent's history is kept to a single orphan commit: every flush
191/// writes a new tree, commits it with no parents, repoints the ref, and
192/// prunes the superseded loose objects. Callers who only need the live set
193/// never see stale gitlinks, and the parent's object DB stays bounded.
194///
195/// # Per-entry repos
196///
197/// Each `modules/<id>.git/` has its own commit chain for that entry alone.
198/// [`Store::put`](crate::Store::put) appends a commit carrying a `label`
199/// blob and/or a `data` blob, [`Store::archive`](crate::Store::archive)
200/// appends a tombstone commit and drops the gitlink from the parent, and
201/// [`Store::history`](crate::Store::history) walks that chain. The
202/// entry-level history is independent of the parent, so pruning gitlinks
203/// never rewrites per-entry history.
204///
205/// # Lazy loading
206///
207/// A module directory only needs to exist on disk when an operation
208/// touches it. `ensure_loaded` materialises an entry's bare repo from
209/// either a preloaded tarball in `pending_modules` or the [`ModuleFetcher`]
210/// installed via [`Store::<SubmoduleLayout>::with_fetcher`], letting a
211/// store back a large entry set without keeping every repo on disk.
212pub struct SubmoduleLayout {
213    path: PathBuf,
214    dirty_parent: bool,
215    dirty_modules: HashMap<EntryId, ModuleDirt>,
216    gitlinks: BTreeMap<EntryId, gix::ObjectId>,
217    label_cache: BTreeMap<EntryId, Vec<u8>>,
218    gitlinks_dirty: bool,
219    pending_modules: Mutex<Modules>,
220    fetcher: Option<ModuleFetcher>,
221}
222
223impl SubmoduleLayout {
224    fn parent_path(&self) -> PathBuf {
225        self.path.join(PARENT_DIR)
226    }
227
228    fn module_path(&self, id: &EntryId) -> PathBuf {
229        self.path
230            .join(MODULES_DIR)
231            .join(format!("{}.git", id.as_str()))
232    }
233
234    fn current_module_commit(&self, id: &EntryId) -> Option<gix::ObjectId> {
235        self.gitlinks.get(id).copied()
236    }
237
238    fn mark_module_changed(&mut self, id: &EntryId) {
239        self.dirty_parent = true;
240        self.dirty_modules.insert(id.clone(), ModuleDirt::Changed);
241    }
242
243    fn mark_module_deleted(&mut self, id: &EntryId) {
244        self.dirty_parent = true;
245        self.dirty_modules.insert(id.clone(), ModuleDirt::Deleted);
246    }
247
248    /// Ensure `id`'s module is extracted to the scratch dir. If the
249    /// dir already exists, no-op. Otherwise consume any pending
250    /// tarball for `id` and untar it. If nothing is pending, fall back
251    /// to the [`ModuleFetcher`] when one was supplied via
252    /// [`Parts::fetcher`]. If neither path produces bytes and the id
253    /// is live in the parent, return an error so the caller knows the
254    /// backing store has diverged from the parent's gitlink set.
255    fn ensure_loaded(&self, id: &EntryId) -> Result<(), Error> {
256        let mod_path = self.module_path(id);
257        if mod_path.exists() {
258            return Ok(());
259        }
260        let bytes = self.pending_modules.lock().unwrap().remove(id);
261        if let Some(bytes) = bytes {
262            untar_into(&bytes, &mod_path)?;
263            return Ok(());
264        }
265        if let Some(fetcher) = &self.fetcher {
266            match fetcher(id).map_err(Error::Fetch)? {
267                Some(bytes) => {
268                    untar_into(&bytes, &mod_path)?;
269                    return Ok(());
270                }
271                None => {
272                    if self.gitlinks.contains_key(id) {
273                        return Err(Error::Other(format!(
274                            "module {} is live in the parent but fetcher returned None",
275                            id.as_str()
276                        )));
277                    }
278                    return Ok(());
279                }
280            }
281        }
282        if self.gitlinks.contains_key(id) {
283            return Err(Error::Other(format!(
284                "module {} is live but its bytes are not loaded; call Store::load_module first",
285                id.as_str()
286            )));
287        }
288        Ok(())
289    }
290
291    /// Materialise the current gitlinks + label_cache into one parent
292    /// commit. No-op when nothing has changed since the last
293    /// materialisation.
294    fn flush_parent(&mut self) -> Result<(), Error> {
295        if !self.gitlinks_dirty {
296            return Ok(());
297        }
298        let parent = gix::open(self.parent_path())?;
299
300        let mut entries: Vec<TreeEntry> = self
301            .gitlinks
302            .iter()
303            .map(|(id, oid)| TreeEntry {
304                mode: EntryKind::Commit.into(),
305                filename: BString::from(id.as_str()),
306                oid: *oid,
307            })
308            .collect();
309
310        if !self.gitlinks.is_empty() {
311            let manifest = serialize_gitmodules(&self.gitlinks);
312            let blob_id = parent.write_blob(&manifest)?.detach();
313            entries.push(TreeEntry {
314                mode: EntryKind::Blob.into(),
315                filename: BString::from(GITMODULES_FILE),
316                oid: blob_id,
317            });
318        }
319
320        if !self.label_cache.is_empty() {
321            let mut index_entries: Vec<TreeEntry> = Vec::with_capacity(self.label_cache.len());
322            for (id, bytes) in &self.label_cache {
323                let blob_id = parent.write_blob(bytes)?.detach();
324                index_entries.push(TreeEntry {
325                    mode: EntryKind::Blob.into(),
326                    filename: BString::from(id.as_str()),
327                    oid: blob_id,
328                });
329            }
330            let index_tree = Tree {
331                entries: index_entries,
332            };
333            let index_tree_id = parent.write_object(&index_tree)?.detach();
334            entries.push(TreeEntry {
335                mode: EntryKind::Tree.into(),
336                filename: BString::from(INDEX_DIR),
337                oid: index_tree_id,
338            });
339        }
340        entries.sort_by(|a, b| a.filename.cmp(&b.filename));
341
342        let tree = Tree { entries };
343        self.commit_parent_tree(&parent, tree)?;
344        self.gitlinks_dirty = false;
345        self.dirty_parent = true;
346        Ok(())
347    }
348
349    /// Write `tree` and a new orphan commit pointing at it into the
350    /// parent's own object DB, then overwrite the parent's
351    /// `refs/heads/main` to point at that commit. Prunes the loose
352    /// objects that the new commit supersedes.
353    fn commit_parent_tree(&self, parent: &gix::Repository, tree: Tree) -> Result<(), Error> {
354        let superseded = self.superseded_parent_objects(parent)?;
355        let tree_id = parent.write_object(&tree)?.detach();
356        let commit_id = write_commit(parent, tree_id, Vec::new(), "parent")?;
357        write_ref_file(&module_ref_path(&self.parent_path()), commit_id)?;
358
359        if let Some((old_commit, old_tree)) = superseded {
360            if old_commit != commit_id {
361                drop_loose_object(&self.parent_path(), old_commit)?;
362            }
363            if let Some(old_tree) = old_tree
364                && old_tree != tree_id
365            {
366                drop_loose_object(&self.parent_path(), old_tree)?;
367            }
368        }
369        Ok(())
370    }
371
372    fn superseded_parent_objects(
373        &self,
374        parent: &gix::Repository,
375    ) -> Result<Option<(gix::ObjectId, Option<gix::ObjectId>)>, Error> {
376        let Some(commit_oid) = read_ref_file(&module_ref_path(&self.parent_path()))? else {
377            return Ok(None);
378        };
379        let tree_oid = parent
380            .find_object(commit_oid)
381            .ok()
382            .and_then(|o| o.try_into_commit().ok())
383            .and_then(|c| c.decode().ok().map(|d| d.tree()));
384        Ok(Some((commit_oid, tree_oid)))
385    }
386}

388impl Layout for SubmoduleLayout {
389    fn new(path: PathBuf) -> Result<Self, Error> {
390        if path.exists() {
391            return Err(Error::Other(format!(
392                "submodule new: path {path:?} already exists"
393            )));
394        }
395        std::fs::create_dir(&path)?;
396        std::fs::create_dir(path.join(MODULES_DIR))?;
397        init_bare_on_branch(&path.join(PARENT_DIR))?;
398        Ok(SubmoduleLayout {
399            path,
400            dirty_parent: true,
401            dirty_modules: HashMap::new(),
402            gitlinks: BTreeMap::new(),
403            label_cache: BTreeMap::new(),
404            gitlinks_dirty: false,
405            pending_modules: Mutex::new(Modules::new()),
406            fetcher: None,
407        })
408    }
409
410    fn open(path: PathBuf) -> Result<Self, Error> {
411        validate_submodule_repo(&path)?;
412        let parent_path = path.join(PARENT_DIR);
413        let modules_path = path.join(MODULES_DIR);
414        if !modules_path.exists() {
415            std::fs::create_dir(&modules_path)?;
416        }
417        let (gitlinks, label_cache) = load_parent_state(&parent_path)?;
418        Ok(SubmoduleLayout {
419            path,
420            dirty_parent: false,
421            dirty_modules: HashMap::new(),
422            gitlinks,
423            label_cache,
424            gitlinks_dirty: false,
425            pending_modules: Mutex::new(Modules::new()),
426            fetcher: None,
427        })
428    }
429
430    fn save(&mut self) -> Result<Vec<u8>, Error> {
431        self.flush_parent()?;
432        let live: Vec<EntryId> = self.gitlinks.keys().cloned().collect();
433        for id in &live {
434            self.ensure_loaded(id)?;
435        }
436        tar_dir(&self.path)
437    }
438
439    fn put(
440        &mut self,
441        id: &EntryId,
442        label: Option<&[u8]>,
443        data: Option<&[u8]>,
444    ) -> Result<Option<CommitId>, Error> {
445        if label.is_none() && data.is_none() {
446            return Err(Error::Other(
447                "Store::put requires at least one of label or data; use Store::archive for a tombstone"
448                    .into(),
449            ));
450        }
451        self.ensure_loaded(id)?;
452        let mod_path = self.module_path(id);
453        if !mod_path.exists() {
454            init_bare_on_branch(&mod_path)?;
455        }
456        let module = gix::open(&mod_path)?;
457        let Some(module_commit) = write_entry_commit(&module, label, data)? else {
458            return Ok(None);
459        };
460        self.gitlinks.insert(id.clone(), module_commit);
461        match label {
462            Some(bytes) if !bytes.is_empty() => {
463                self.label_cache.insert(id.clone(), bytes.to_vec());
464            }
465            Some(_) => {
466                self.label_cache.remove(id);
467            }
468            None => {}
469        }
470        self.gitlinks_dirty = true;
471        self.mark_module_changed(id);
472        Ok(Some(module_commit.into()))
473    }
474
475    fn get(&self, id: &EntryId) -> Result<Option<Entry>, Error> {
476        let Some(commit) = self.current_module_commit(id) else {
477            return Ok(None);
478        };
479        self.ensure_loaded(id)?;
480        let repo = gix::open(self.module_path(id))?;
481        Ok(Some(read_entry_at(&repo, commit)?))
482    }
483
484    fn archive(&mut self, id: &EntryId) -> Result<(), Error> {
485        if !self.gitlinks.contains_key(id) {
486            return Ok(());
487        }
488        self.ensure_loaded(id)?;
489        let module = gix::open(self.module_path(id))?;
490        write_tombstone_commit(&module)?;
491        self.gitlinks.remove(id);
492        self.label_cache.remove(id);
493        self.gitlinks_dirty = true;
494        self.mark_module_changed(id);
495        Ok(())
496    }
497
498    fn delete(&mut self, id: &EntryId) -> Result<(), Error> {
499        let mod_path = self.module_path(id);
500        if mod_path.exists() {
501            std::fs::remove_dir_all(&mod_path)?;
502        }
503        self.pending_modules.lock().unwrap().remove(id);
504        if self.gitlinks.remove(id).is_some() {
505            self.label_cache.remove(id);
506            self.gitlinks_dirty = true;
507            self.mark_module_deleted(id);
508        }
509        Ok(())
510    }
511
512    fn list(&self) -> Result<Vec<EntryId>, Error> {
513        Ok(self.gitlinks.keys().cloned().collect())
514    }
515
516    fn history(&self, id: &EntryId) -> Result<Vec<Entry>, Error> {
517        self.ensure_loaded(id)?;
518        let mod_path = self.module_path(id);
519        if !mod_path.exists() {
520            return Ok(Vec::new());
521        }
522        let repo = gix::open(&mod_path)?;
523        let Ok(head) = repo.head_commit() else {
524            return Ok(Vec::new());
525        };
526        let mut out = Vec::new();
527        for info in head.ancestors().all()? {
528            let info = info?;
529            out.push(read_entry_at(&repo, info.id)?);
530        }
531        Ok(out)
532    }
533
534    fn label(&self, id: &EntryId) -> Option<&[u8]> {
535        self.label_cache.get(id).map(Vec::as_slice)
536    }
537
538    fn list_labels(&self) -> Vec<(EntryId, Vec<u8>)> {
539        self.label_cache
540            .iter()
541            .map(|(k, v)| (k.clone(), v.clone()))
542            .collect()
543    }
544}

546impl Drop for SubmoduleLayout {
547    /// Materialise any pending parent commit to disk. Per-module
548    /// commits are already persistent (every `put` writes the commit
549    /// and updates the module's ref before returning), so only the
550    /// gitlink-set / label-cache batched in memory needs flushing.
551    ///
552    /// Errors are swallowed -- Drop can't surface them. Callers who
553    /// need error-handling should call `snapshot` or `save`
554    /// explicitly.
555    fn drop(&mut self) {
556        let _ = self.flush_parent();
557    }
558}

560/// Sanity-check that `path` holds a valid submodule-layout storgit
561/// repo: a directory containing `parent.git/` as a bare repo whose
562/// HEAD points to storgit's branch. `modules/` may or may not exist
563/// yet (it's created on demand), so its presence is not required.
564fn validate_submodule_repo(path: &std::path::Path) -> Result<(), Error> {
565    if !path.is_dir() {
566        return Err(Error::Other(format!(
567            "submodule open: {path:?} is not a directory"
568        )));
569    }
570    let parent_path = path.join(PARENT_DIR);
571    if !parent_path.is_dir() {
572        return Err(Error::Other(format!(
573            "submodule open: {parent_path:?} does not exist"
574        )));
575    }
576    gix::open(&parent_path).map_err(|e| {
577        Error::Other(format!(
578            "submodule open: {parent_path:?} is not a git repo: {e}"
579        ))
580    })?;
581    let head_raw = std::fs::read_to_string(parent_path.join("HEAD"))
582        .map_err(|e| Error::Other(format!("submodule open: cannot read HEAD: {e}")))?;
583    let head_trimmed = head_raw.trim();
584    let expected = format!("ref: {BRANCH}");
585    if head_trimmed != expected {
586        return Err(Error::Other(format!(
587            "submodule open: HEAD must be {expected:?}; got {head_trimmed:?}"
588        )));
589    }
590    Ok(())
591}

595impl Store<SubmoduleLayout> {
596    /// Fold [`Parts`] into an opened submodule-layout store.
597    ///
598    /// - `parts.parent`: if non-empty, untarred into `parent.git`,
599    ///   replacing whatever `open` produced. Requires the existing
600    ///   parent to be empty (freshly initialised). TODO: once
601    ///   merging lands (see PLAN-MERGE.md), allow applying parent
602    ///   bytes onto an existing non-empty parent by merging.
603    /// - `parts.modules`: inserted into the pending-modules cache;
604    ///   each id's tarball is untarred on first touch.
605    pub fn with_parts(mut self, parts: Parts) -> Result<Self, Error> {
606        if !parts.parent.is_empty() {
607            // TODO: merge parts.parent with the on-disk parent when
608            // the latter already has history. For now, require the
609            // on-disk parent to be freshly init'd (dirty_parent is
610            // the in-memory signal for that state).
611            if !self.layout.dirty_parent {
612                return Err(Error::Other(
613                    "with_parts: parts.parent is non-empty but the store's parent.git already \
614                     has state; merging is not yet implemented"
615                        .into(),
616                ));
617            }
618            let parent_path = self.layout.parent_path();
619            if parent_path.exists() {
620                std::fs::remove_dir_all(&parent_path)?;
621            }
622            untar_into(&parts.parent, &parent_path)?;
623            let (gitlinks, label_cache) = load_parent_state(&parent_path)?;
624            self.layout.gitlinks = gitlinks;
625            self.layout.label_cache = label_cache;
626            self.layout.dirty_parent = false;
627        }
628        let pending = self.layout.pending_modules.get_mut().unwrap();
629        for (id, bytes) in parts.modules {
630            pending.insert(id, bytes);
631        }
632        Ok(self)
633    }

635    /// Install a [`ModuleFetcher`] as a lazy fallback for module
636    /// bytes, replacing any prior fetcher. See [`ModuleFetcher`] for
637    /// the contract.
638    pub fn with_fetcher(mut self, fetcher: ModuleFetcher) -> Self {
639        self.layout.fetcher = Some(fetcher);
640        self
641    }

643    /// Re-tar everything touched since the previous snapshot (or since
644    /// [`Store::<SubmoduleLayout>::open`] for the first call) and hand
645    /// the caller exactly the parts that need repersisting. Clears
646    /// dirty tracking on success, so back-to-back snapshots with no
647    /// intervening writes return an empty [`Snapshot`].
648    pub fn snapshot(&mut self) -> Result<Snapshot, Error> {
649        self.layout.flush_parent()?;
650        let mut snap = Snapshot::default();
651        if self.layout.dirty_parent {
652            snap.parent = Some(tar_dir(&self.layout.parent_path())?);
653        }
654        let dirty = std::mem::take(&mut self.layout.dirty_modules);
655        for (name, state) in dirty {
656            let change = match state {
657                ModuleDirt::Changed => {
658                    let bytes = tar_dir(&self.layout.module_path(&name))?;
659                    ModuleChange::Changed(bytes)
660                }
661                ModuleDirt::Deleted => ModuleChange::Deleted,
662            };
663            snap.modules.insert(name, change);
664        }
665        self.layout.dirty_parent = false;
666        Ok(snap)
667    }
668
669    /// Make `bytes` (a previously-persisted module tarball) available
670    /// to the store under `id`. The next operation that touches `id`
671    /// will untar these bytes into the store's path on first access.
672    pub fn load_module(&mut self, id: EntryId, bytes: Vec<u8>) {
673        self.layout
674            .pending_modules
675            .get_mut()
676            .unwrap()
677            .insert(id, bytes);
678    }
