24/// Subtree name inside the repo's root tree that holds every entry.
25const RECORDS_DIR: &str = "records";
26
27/// Backed by a single bare repo at `path`. Every entry lives as a
28/// subtree of that repo's root tree, and a single shared ref
29/// advances with every write.
30///
31/// # Disk layout
32///
33/// ```text
34/// <path>/                 bare repo
35///   HEAD -> refs/heads/storgit
36///   refs/heads/storgit    -> the latest commit
37///   objects/              the shared object DB for every entry
38/// ```
39///
40/// # Tree shape
41///
42/// The tip commit's root tree has a single `records/` subtree. Inside
43/// `records/`, each live entry is a subtree named by its [`EntryId`]
44/// carrying up to two blobs:
45///
46/// - `data`, the entry's payload bytes
47/// - `label`, the entry's label bytes (omitted when the label is empty)
48///
49/// ```text
50/// <root>
51///   records/
52///     <id-a>/
53///       data
54///       label
55///     <id-b>/
56///       data
57/// ```
58///
59/// `DATA_FILE` sorts before `LABEL_FILE`, so entry subtrees are
60/// assembled in that order to satisfy git's strict filename sort
61/// without an explicit sort call.
62///
63/// # Write path
64///
65/// Each `put` or `archive` builds a new `records/<id>/` subtree,
66/// splices it into a new `records/` tree (copying the other ids
67/// unchanged), wraps that in a new root tree, and commits with the
68/// prior tip as parent. The new commit is published by rewriting
69/// `refs/heads/storgit` to point at it. `put` is a no-op when the
70/// rebuilt `records/<id>/` subtree is byte-identical to the prior
71/// one; in that case no commit is written and the ref is unchanged.
72/// `delete` is currently an alias for `archive`: a single shared ref
73/// can't cheaply excise one entry's history without rewriting every
74/// commit that touched it.
75///
76/// # Per-entry history
77///
78/// There are no per-entry refs. [`Store::history`](crate::Store::history)
79/// walks the shared commit graph and emits an [`Entry`] for each
80/// commit where the `records/<id>/` subtree differs from its parent's,
81/// which mirrors what `git log -- records/<id>/` would surface. The
82/// walk reads the subtree oid at each commit and compares it to the
83/// first parent's, so it's O(history length) but each step is a
84/// constant-time oid compare.
85///
86/// # Label cache
87///
88/// [`Layout::label`] and [`Layout::list_labels`] are served from an
89/// in-memory `label_cache` keyed by [`EntryId`]. The cache is populated on
90/// [`Layout::open`] by walking `HEAD:records/` once, and kept current
91/// by `put` and `archive`.
92pub struct SubdirLayout {
93    path: PathBuf,
94    label_cache: BTreeMap<EntryId, Vec<u8>>,
95}

97impl SubdirLayout {
98    fn open_repo(&self) -> Result<gix::Repository, Error> {
99        Ok(gix::open(&self.path)?)
100    }
101
102    fn head_commit(&self, _repo: &gix::Repository) -> Result<Option<gix::ObjectId>, Error> {
103        read_ref_file(&module_ref_path(&self.path))
104    }
105
106    /// Tree oid of `records/<id>/` at the given root tree, if any.
107    fn id_subtree_oid(
108        repo: &gix::Repository,
109        root_tree_id: gix::ObjectId,
110        id: &EntryId,
111    ) -> Result<Option<gix::ObjectId>, Error> {
112        let Some(records_oid) = subtree_entry(repo, root_tree_id, RECORDS_DIR)? else {
113            return Ok(None);
114        };
115        subtree_entry(repo, records_oid, id.as_str())
116    }
117
118    /// Build and persist the label_cache by walking `HEAD:records/`.
119    fn rebuild_label_cache(&mut self) -> Result<(), Error> {
120        self.label_cache.clear();
121        let repo = self.open_repo()?;
122        let Some(head) = self.head_commit(&repo)? else {
123            return Ok(());
124        };
125        let commit = repo.find_object(head)?.into_commit();
126        let root_tree_id = commit.decode()?.tree();
127        let Some(records_oid) = subtree_entry(&repo, root_tree_id, RECORDS_DIR)? else {
128            return Ok(());
129        };
130        let records_tree = decode_tree(&repo, records_oid)?;
131        for entry in records_tree.entries {
132            if !matches!(entry.mode.kind(), EntryKind::Tree) {
133                continue;
134            }
135            let id = EntryId::new(entry.filename.to_string()).map_err(|e| {
136                Error::Other(format!("corrupt records/ entry {:?}: {e}", entry.filename))
137            })?;
138            let id_tree = decode_tree(&repo, entry.oid)?;
139            if let Some(label_entry) = id_tree
140                .entries
141                .iter()
142                .find(|e| e.filename.as_bstr() == BStr::new(LABEL_FILE))
143            {
144                let blob = repo.find_object(label_entry.oid)?;
145                if !blob.data.is_empty() {
146                    self.label_cache.insert(id, blob.data.clone());
147                }
148            }
149        }
150        Ok(())
151    }
152}

154/// Look up a named child of `tree_id` and return its oid, or `None`
155/// if that child doesn't exist. Matches by byte-wise filename.
156fn subtree_entry(
157    repo: &gix::Repository,
158    tree_id: gix::ObjectId,
159    name: &str,
160) -> Result<Option<gix::ObjectId>, Error> {
161    let tree = decode_tree(repo, tree_id)?;
162    Ok(tree
163        .entries
164        .into_iter()
165        .find(|e| e.filename.as_bstr() == BStr::new(name))
166        .map(|e| e.oid))
167}

169impl Layout for SubdirLayout {
170    fn new(path: PathBuf) -> Result<Self, Error> {
171        if path.exists() {
172            return Err(Error::Other(format!(
173                "subdir new: path {path:?} already exists"
174            )));
175        }
176        std::fs::create_dir(&path)?;
177        init_bare_on_branch(&path)?;
178        Ok(SubdirLayout {
179            path,
180            label_cache: BTreeMap::new(),
181        })
182    }
183
184    fn open(path: PathBuf) -> Result<Self, Error> {
185        validate_subdir_repo(&path)?;
186        let mut layout = SubdirLayout {
187            path,
188            label_cache: BTreeMap::new(),
189        };
190        layout.rebuild_label_cache()?;
191        Ok(layout)
192    }
193
194    fn save(&mut self) -> Result<Vec<u8>, Error> {
195        tar_dir(&self.path)
196    }
197
198    fn put(
199        &mut self,
200        id: &EntryId,
201        label: Option<&[u8]>,
202        data: Option<&[u8]>,
203    ) -> Result<Option<CommitId>, Error> {
204        if label.is_none() && data.is_none() {
205            return Err(Error::Other(
206                "Store::put requires at least one of label or data; use Store::archive for a tombstone"
207                    .into(),
208            ));
209        }
210        let repo = self.open_repo()?;
211        let prior_commit = self.head_commit(&repo)?;
212        let prior_root_tree_id = if let Some(pid) = prior_commit {
213            Some(repo.find_object(pid)?.into_commit().decode()?.tree())
214        } else {
215            None
216        };
217        let prior_records_oid = match prior_root_tree_id {
218            Some(tid) => subtree_entry(&repo, tid, RECORDS_DIR)?,
219            None => None,
220        };
221        let prior_id_subtree_oid = match prior_records_oid {
222            Some(roid) => subtree_entry(&repo, roid, id.as_str())?,
223            None => None,
224        };
225        let prior_id_entries = match prior_id_subtree_oid {
226            Some(sid) => Some(decode_tree(&repo, sid)?.entries),
227            None => None,
228        };
229        let prior_blob = |filename: &str| -> Option<gix::ObjectId> {
230            prior_id_entries.as_ref().and_then(|entries| {
231                entries
232                    .iter()
233                    .find(|e| e.filename.as_bstr() == BStr::new(filename))
234                    .map(|e| e.oid)
235            })
236        };
237
238        let data_oid = match data {
239            Some(bytes) => Some(repo.write_blob(bytes)?.detach()),
240            None => prior_blob(DATA_FILE),
241        };
242        let label_oid = match label {
243            Some(bytes) => Some(repo.write_blob(bytes)?.detach()),
244            None => prior_blob(LABEL_FILE),
245        };
246
247        let mut id_entries: Vec<TreeEntry> = Vec::with_capacity(2);
248        if let Some(oid) = data_oid {
249            id_entries.push(TreeEntry {
250                mode: EntryKind::Blob.into(),
251                filename: BString::from(DATA_FILE),
252                oid,
253            });
254        }
255        if let Some(oid) = label_oid {
256            id_entries.push(TreeEntry {
257                mode: EntryKind::Blob.into(),
258                filename: BString::from(LABEL_FILE),
259                oid,
260            });
261        }
262        // DATA_FILE < LABEL_FILE lexicographically, so push order already
263        // satisfies git's strict filename sort.
264        let id_tree = Tree {
265            entries: id_entries,
266        };
267        let id_tree_id = repo.write_object(&id_tree)?.detach();
268
269        // No-op detection: identical id-subtree means identical commit
270        // for this record.
271        if prior_id_subtree_oid == Some(id_tree_id) {
272            return Ok(None);
273        }
274
275        let records_tree_id = write_records_tree(&repo, prior_records_oid, id, Some(id_tree_id))?;
276        let root_tree_id = write_root_tree(&repo, prior_root_tree_id, records_tree_id)?;
277        let commit_id = write_commit(
278            &repo,
279            root_tree_id,
280            prior_commit.into_iter().collect(),
281            "put",
282        )?;
283        write_ref_file(&module_ref_path(&self.path), commit_id)?;
284
285        match label {
286            Some(bytes) if !bytes.is_empty() => {
287                self.label_cache.insert(id.clone(), bytes.to_vec());
288            }
289            Some(_) => {
290                self.label_cache.remove(id);
291            }
292            None => {}
293        }
294        Ok(Some(commit_id.into()))
295    }
296
297    fn get(&self, id: &EntryId) -> Result<Option<Entry>, Error> {
298        let repo = self.open_repo()?;
299        let Some(head) = self.head_commit(&repo)? else {
300            return Ok(None);
301        };
302        let commit = repo.find_object(head)?.into_commit();
303        let root_tree_id = commit.decode()?.tree();
304        if Self::id_subtree_oid(&repo, root_tree_id, id)?.is_none() {
305            return Ok(None);
306        }
307        Ok(Some(read_subdir_entry_at(&repo, head, id)?))
308    }
309
310    fn archive(&mut self, id: &EntryId) -> Result<(), Error> {
311        let repo = self.open_repo()?;
312        let Some(prior_commit) = self.head_commit(&repo)? else {
313            return Ok(());
314        };
315        let prior_root_tree_id = repo
316            .find_object(prior_commit)?
317            .into_commit()
318            .decode()?
319            .tree();
320        let prior_records_oid = subtree_entry(&repo, prior_root_tree_id, RECORDS_DIR)?;
321        let Some(roid) = prior_records_oid else {
322            return Ok(());
323        };
324        if subtree_entry(&repo, roid, id.as_str())?.is_none() {
325            return Ok(());
326        }
327        let records_tree_id = write_records_tree(&repo, Some(roid), id, None)?;
328        let root_tree_id = write_root_tree(&repo, Some(prior_root_tree_id), records_tree_id)?;
329        let commit_id = write_commit(&repo, root_tree_id, vec![prior_commit], "archive")?;
330        write_ref_file(&module_ref_path(&self.path), commit_id)?;
331        self.label_cache.remove(id);
332        Ok(())
333    }
334
335    fn delete(&mut self, id: &EntryId) -> Result<(), Error> {
336        // A single-ref layout can't cheaply erase history for one
337        // record without rewriting the ref. For now `delete` behaves
338        // like `archive`: the entry is removed from the tree, and the
339        // past commits that touched it remain reachable from HEAD.
340        self.archive(id)
341    }
342
343    fn list(&self) -> Result<Vec<EntryId>, Error> {
344        let repo = self.open_repo()?;
345        let Some(head) = self.head_commit(&repo)? else {
346            return Ok(Vec::new());
347        };
348        let commit = repo.find_object(head)?.into_commit();
349        let root_tree_id = commit.decode()?.tree();
350        let Some(records_oid) = subtree_entry(&repo, root_tree_id, RECORDS_DIR)? else {
351            return Ok(Vec::new());
352        };
353        let records_tree = decode_tree(&repo, records_oid)?;
354        let mut out = Vec::with_capacity(records_tree.entries.len());
355        for entry in records_tree.entries {
356            if !matches!(entry.mode.kind(), EntryKind::Tree) {
357                continue;
358            }
359            let id = EntryId::new(entry.filename.to_string()).map_err(|e| {
360                Error::Other(format!("corrupt records/ entry {:?}: {e}", entry.filename))
361            })?;
362            out.push(id);
363        }
364        Ok(out)
365    }
366
367    fn history(&self, id: &EntryId) -> Result<Vec<Entry>, Error> {
368        let repo = self.open_repo()?;
369        let Some(head) = self.head_commit(&repo)? else {
370            return Ok(Vec::new());
371        };
372        let head_obj = repo.find_object(head)?.into_commit();
373        let mut out = Vec::new();
374        for info in head_obj.ancestors().all()? {
375            let info = info?;
376            let commit = repo.find_object(info.id)?.into_commit();
377            let decoded = commit.decode()?;
378            let tree_id = decoded.tree();
379            let this_subtree = Self::id_subtree_oid(&repo, tree_id, id)?;
380            let parent_ids: Vec<gix::ObjectId> = decoded.parents().collect();
381            let parent_subtree = if let Some(&pid) = parent_ids.first() {
382                let pc = repo.find_object(pid)?.into_commit();
383                let ptid = pc.decode()?.tree();
384                Self::id_subtree_oid(&repo, ptid, id)?
385            } else {
386                None
387            };
388            if this_subtree != parent_subtree {
389                out.push(read_subdir_entry_at(&repo, info.id, id)?);
390            }
391        }
392        Ok(out)
393    }
394
395    fn label(&self, id: &EntryId) -> Option<&[u8]> {
396        self.label_cache.get(id).map(Vec::as_slice)
397    }
398
399    fn list_labels(&self) -> Vec<(EntryId, Vec<u8>)> {
400        self.label_cache
401            .iter()
402            .map(|(k, v)| (k.clone(), v.clone()))
403            .collect()
404    }
405}

407/// Build a new `records/` subtree by copying the entries of the prior
408/// records subtree (if any), removing or replacing the entry for
409/// `id`, and writing the result. `new_id_subtree` of `None` removes
410/// the id; `Some(oid)` replaces or inserts it.
411fn write_records_tree(
412    repo: &gix::Repository,
413    prior_records_oid: Option<gix::ObjectId>,
414    id: &EntryId,
415    new_id_subtree: Option<gix::ObjectId>,
416) -> Result<Option<gix::ObjectId>, Error> {
417    let mut entries: Vec<TreeEntry> = match prior_records_oid {
418        Some(roid) => decode_tree(repo, roid)?
419            .entries
420            .into_iter()
421            .filter(|e| e.filename.as_bstr() != BStr::new(id.as_str()))
422            .collect(),
423        None => Vec::new(),
424    };
425    if let Some(oid) = new_id_subtree {
426        entries.push(TreeEntry {
427            mode: EntryKind::Tree.into(),
428            filename: BString::from(id.as_str()),
429            oid,
430        });
431    }
432    if entries.is_empty() {
433        return Ok(None);
434    }
435    entries.sort_by(|a, b| a.filename.cmp(&b.filename));
436    let tree = Tree { entries };
437    Ok(Some(repo.write_object(&tree)?.detach()))
438}

440/// Build the new root tree by copying prior root entries (if any),
441/// removing or replacing the `records/` entry, and writing the
442/// result. `new_records` of `None` drops the `records/` subtree.
443fn write_root_tree(
444    repo: &gix::Repository,
445    prior_root_tree_id: Option<gix::ObjectId>,
446    new_records: Option<gix::ObjectId>,
447) -> Result<gix::ObjectId, Error> {
448    let mut entries: Vec<TreeEntry> = match prior_root_tree_id {
449        Some(tid) => decode_tree(repo, tid)?
450            .entries
451            .into_iter()
452            .filter(|e| e.filename.as_bstr() != BStr::new(RECORDS_DIR))
453            .collect(),
454        None => Vec::new(),
455    };
456    if let Some(oid) = new_records {
457        entries.push(TreeEntry {
458            mode: EntryKind::Tree.into(),
459            filename: BString::from(RECORDS_DIR),
460            oid,
461        });
462    }
463    entries.sort_by(|a, b| a.filename.cmp(&b.filename));
464    let tree = Tree { entries };
465    Ok(repo.write_object(&tree)?.detach())
466}

468/// Read a single [`Entry`] for `id` at commit `commit_id`. If the
469/// `records/<id>/` subtree is absent at that commit (the archive
470/// point), returns an entry with both slots `None`.
471fn read_subdir_entry_at(
472    repo: &gix::Repository,
473    commit_id: gix::ObjectId,
474    id: &EntryId,
475) -> Result<Entry, Error> {
476    let commit = repo.find_object(commit_id)?.into_commit();
477    let sig = commit.committer()?;
478    let seconds = sig.seconds().max(0) as u64;
479    let time = std::time::UNIX_EPOCH + std::time::Duration::from_secs(seconds);
480    let root_tree_id = commit.decode()?.tree();
481    let id_subtree_oid = SubdirLayout::id_subtree_oid(repo, root_tree_id, id)?;
482    let (label, data) = match id_subtree_oid {
483        Some(sid) => {
484            let id_tree = decode_tree(repo, sid)?;
485            let mut label = None;
486            let mut data = None;
487            for entry in id_tree.entries {
488                if entry.filename.as_bstr() == BStr::new(DATA_FILE) {
489                    data = Some(repo.find_object(entry.oid)?.data.clone());
490                } else if entry.filename.as_bstr() == BStr::new(LABEL_FILE) {
491                    label = Some(repo.find_object(entry.oid)?.data.clone());
492                }
493            }
494            (label, data)
495        }
496        None => (None, None),
497    };
498    Ok(Entry {
499        commit: commit_id.into(),
500        time,
501        label,
502        data,
503    })
504}

506/// Sanity-check that `path` holds a valid subdir-layout storgit
507/// repo: a bare repo whose HEAD points to storgit's branch and
508/// whose root tree (if any commit exists) contains no gitlinks
509/// (which would signal a submodule-layout repo).
510fn validate_subdir_repo(path: &Path) -> Result<(), Error> {
511    let repo = gix::open(path).map_err(|e| {
512        Error::Other(format!(
513            "storgit subdir open: path {path:?} is not a git repo: {e}"
514        ))
515    })?;
516    let head_raw = std::fs::read_to_string(path.join("HEAD"))
517        .map_err(|e| Error::Other(format!("storgit subdir open: cannot read HEAD: {e}")))?;
518    let head_trimmed = head_raw.trim();
519    let expected = format!("ref: {BRANCH}");
520    if head_trimmed != expected {
521        return Err(Error::Other(format!(
522            "storgit subdir open: HEAD must be {expected:?}; got {head_trimmed:?}"
523        )));
524    }
525    if let Some(commit_oid) = read_ref_file(&module_ref_path(path))? {
526        let commit = repo.find_object(commit_oid)?.into_commit();
527        let tree_id = commit.decode()?.tree();
528        let tree = decode_tree(&repo, tree_id)?;
529        for entry in tree.entries {
530            if matches!(entry.mode.kind(), EntryKind::Commit) {
531                return Err(Error::Other(format!(
532                    "storgit subdir open: {path:?} looks like a submodule-layout storgit repo (gitlinks in root tree)"
533                )));
534            }
535        }
536    }
537    Ok(())
538}
