1//! On-disk shapes that back a [`Store`](crate::Store).
2//!
3//! The [`Layout`] trait is the surface API that every concrete storgit layout
4//! (submodule, subdir) implements. A [`Store`] is a thin generic wrapper over
5//! a layout and just delegates each method to the layout. Available layouts
6//! are:
7//!
8//! - [`SubdirLayout`]: a single bare repo at `<path>/` whose tree carries all
9//! entries as `<path>/records/<id>/{data,label}`, with one shared ref and
10//! per-entry history via path-scoped walks.
11//! - [`SubmoduleLayout`]: a `<path>/parent.git/` bare repo whose tree carries
12//! one _gitlink_ per live entry plus a `.gitmodules` manifest, alongside
13//! `<path>/modules/<id>/{data,label}` bare repos each holding that entry's
14//! `{data,label}` blobs in their own object database.
15//!
16//! Operations that are specific to one layout (persistence envelopes like
17//! [`submodule::Parts`] / [`submodule::Snapshot`], etc.) stay as inherent
18//! methods on that layout's `Store<L>` rather than living on this trait.
19//!
20//! [`Store`]: crate::Store
21//! [`SubdirLayout`]: crate::SubdirLayout
22//! [`SubmoduleLayout`]: crate::SubmoduleLayout
23
24use std::path::PathBuf;
25
26use crate::entry::{CommitId, Entry};
27use crate::error::Error;
28use crate::id::Id;
29use crate::tarball::untar_into;
30
31pub mod subdir;
32pub mod submodule;
33
34pub trait Layout: Sized {
35    /// Initialize a fresh storgit repo at `path`. Creates `path` as
36    /// a new directory; errors if `path` already exists. The parent
37    /// directory must already exist -- `new` does not create
38    /// ancestors.
39    fn new(path: PathBuf) -> Result<Self, Error>;
40
41    /// Open an existing storgit repo at `path`. Errors if `path`
42    /// does not exist or is not a valid storgit repo for this layout.
43    fn open(path: PathBuf) -> Result<Self, Error>;
44
45    /// Bundle the store into a single self-contained tarball. See
46    /// per-layout docs for what "self-contained" means (e.g.
47    /// submodule force-loads every live module before tarring).
48    fn save(&mut self) -> Result<Vec<u8>, Error>;
49
50    /// Untar `bytes` into `path`, then open the result. `path` must
51    /// not exist or must be empty. Default impl works for any layout
52    /// whose on-disk shape is what `save` produced.
53    fn load(bytes: &[u8], path: PathBuf) -> Result<Self, Error> {
54        if path.exists() && path.read_dir()?.next().is_some() {
55            return Err(Error::Other(format!(
56                "load: target path {path:?} is not empty"
57            )));
58        }
59        untar_into(bytes, &path)?;
60        Self::open(path)
61    }
62
63    /// Write a new version of entry `id` carrying the given `label`
64    /// and/or `data` blobs.
65    ///
66    /// At least one of `label` or `data` must be `Some`; passing
67    /// `(None, None)` returns an error. Use [`Layout::archive`] to
68    /// record a tombstone.
69    ///
70    /// Slot semantics:
71    /// - `Some(bytes)` replaces that slot with `bytes` (empty slice
72    ///   is allowed and is distinct from absent).
73    /// - `None` carries the prior commit's blob forward unchanged;
74    ///   on a fresh entry, the slot is simply omitted.
75    ///
76    /// No-op detection: if the resulting per-entry tree is identical
77    /// to the prior one, no commit is written and `Ok(None)` is
78    /// returned. Otherwise returns the new [`CommitId`].
79    fn put(
80        &mut self,
81        id: &Id,
82        label: Option<&[u8]>,
83        data: Option<&[u8]>,
84    ) -> Result<Option<CommitId>, Error>;
85
86    /// Return the latest [`Entry`] for `id`, or `None` if `id` is not
87    /// currently live (never written, archived, or deleted).
88    fn get(&self, id: &Id) -> Result<Option<Entry>, Error>;
89
90    /// Soft-delete `id`: record a tombstone commit so the entry stops
91    /// appearing in [`Layout::list`] / [`Layout::get`], but its prior
92    /// versions remain reachable via [`Layout::history`]. Archiving
93    /// an unknown or already-archived id is a no-op.
94    fn archive(&mut self, id: &Id) -> Result<(), Error>;
95
96    /// Hard-delete `id`: remove the entry and, where the layout
97    /// supports it, its history as well. Layouts that cannot cheaply
98    /// erase per-entry history may fall back to the same behaviour
99    /// as [`Layout::archive`]; see the per-layout docs. Deleting an
100    /// unknown id is a no-op.
101    fn delete(&mut self, id: &Id) -> Result<(), Error>;
102
103    /// List the ids of all live entries, in arbitrary order.
104    /// Archived and deleted ids are excluded.
105    fn list(&self) -> Result<Vec<Id>, Error>;
106
107    /// Walk every historical version of `id`, newest first. Includes
108    /// the tombstone commit for archived entries. Returns an empty
109    /// vec if `id` has no history in this store.
110    fn history(&self, id: &Id) -> Result<Vec<Entry>, Error>;
111
112    /// Return the current label blob for `id`, or `None` if `id` is
113    /// not live or its label slot is absent/empty. Served from an
114    /// in-memory cache, so this is cheap and does not hit the repo.
115    fn label(&self, id: &Id) -> Option<&[u8]>;
116
117    /// Return every live entry whose label slot is non-empty, paired
118    /// with that label blob. Served from the same in-memory cache as
119    /// [`Layout::label`].
120    fn list_labels(&self) -> Vec<(Id, Vec<u8>)>;
121}
