1//! Unique entry identifiers, used as special directory names.
2//!
3//! [`Id`] is the one type each layout uses to name an entry on disk and in git
4//! tree entries, and its validating constructor is the single chokepoint that
5//! enforces the character and length rules both layouts depend on.
6
7use std::borrow::Borrow;
8use std::str::FromStr;
9
10/// A validated entry identifier. Constructed via [`Id::new`] (or
11/// `s.parse::<Id>()`); the validating constructor enforces every
12/// constraint storgit needs to safely use the id as a filename
13/// (`modules/<id>.git`), as a git tree entry name, and as a key in
14/// the parent tree's gitlink set.
15///
16/// Rules enforced:
17/// - non-empty
18/// - at most [`Id::MAX_LEN`] bytes
19/// - no `/` (would create a subdirectory in `modules/`)
20/// - no `"` or `\` (would need escaping inside the `.gitmodules`
21///   section name storgit writes for plain-`git` interop)
22/// - no ASCII control characters (`< 0x20` or DEL, including `\0`,
23///   `\n`, `\t`); they would corrupt git tree filenames or the
24///   `.gitmodules` config file
25/// - no leading `.` (rejects `.`, `..`, hidden-file ids)
26/// - does not end in `.git` (collides with the `<id>.git` module dir)
27/// - is not a [reserved name](Id::is_reserved) used by storgit
28///   internally (currently just the string `"index"`, which collides
29///   with the parent tree's index subtree)
30///
31/// Holding an [`Id`] value is therefore proof that the bytes are safe
32/// to plug into all of storgit's internal paths and tree writes.
33#[derive(Debug, Clone, PartialEq, Eq, PartialOrd, Ord, Hash)]
34pub struct Id(String);
35
36/// Reasons [`Id::new`] can reject a candidate id.
37#[derive(Debug, Clone, PartialEq, Eq)]
38pub enum Error {
39    Empty,
40    TooLong { len: usize, max: usize },
41    BadChar(char),
42    LeadingDot,
43    GitSuffix,
44    Reserved,
45}
46
47impl Id {
48    /// Maximum byte length of a valid id. Picked to leave headroom
49    /// under typical filesystem name limits (255 bytes on most
50    /// filesystems) once the `<id>.git` suffix is appended.
51    pub const MAX_LEN: usize = 240;
52
53    /// True when `s` is a reserved name storgit uses for its own
54    /// bookkeeping inside the parent tree, and therefore must not be
55    /// used as an entry id.
56    pub fn is_reserved(s: &str) -> bool {
57        s == crate::parent::INDEX_DIR
58    }
59
60    /// Construct a validated id, returning [`Error`] on rejection.
61    pub fn new(s: impl Into<String>) -> Result<Self, Error> {
62        let s = s.into();
63        Self::validate(&s)?;
64        Ok(Id(s))
65    }
66
67    pub fn as_str(&self) -> &str {
68        &self.0
69    }
70
71    fn validate(s: &str) -> Result<(), Error> {
72        if s.is_empty() {
73            return Err(Error::Empty);
74        }
75        if s.len() > Self::MAX_LEN {
76            return Err(Error::TooLong {
77                len: s.len(),
78                max: Self::MAX_LEN,
79            });
80        }
81        if s.starts_with('.') {
82            return Err(Error::LeadingDot);
83        }
84        if s.ends_with(".git") {
85            return Err(Error::GitSuffix);
86        }
87        if Self::is_reserved(s) {
88            return Err(Error::Reserved);
89        }
90        for c in s.chars() {
91            if c == '/' || c == '"' || c == '\\' {
92                return Err(Error::BadChar(c));
93            }
94            let code = c as u32;
95            if code < 0x20 || code == 0x7f {
96                return Err(Error::BadChar(c));
97            }
98        }
99        Ok(())
100    }
101}

103impl std::fmt::Display for Id {
104    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
105        f.write_str(&self.0)
106    }
107}
108
109impl std::fmt::Display for Error {
110    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
111        match self {
112            Error::Empty => write!(f, "id is empty"),
113            Error::TooLong { len, max } => write!(f, "id is {len} bytes; max is {max}"),
114            Error::BadChar(c) => write!(f, "id contains forbidden character {c:?}"),
115            Error::LeadingDot => write!(f, "id may not start with '.'"),
116            Error::GitSuffix => write!(f, "id may not end with '.git'"),
117            Error::Reserved => write!(f, "id is reserved by storgit"),
118        }
119    }
120}
121
122impl std::error::Error for Error {}
