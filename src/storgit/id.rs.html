6/// A validated entry identifier.
7///
8/// Constructed via [`EntryId::new`] (or `s.parse::<EntryId>()`); the
9/// validating constructor enforces every constraint storgit needs to
10/// safely use the id as a filename (`modules/<id>.git`), as a git tree
11/// entry name, and as a key in the parent tree's gitlink set.
12///
13/// Rules enforced:
14/// - non-empty
15/// - at most [`EntryId::MAX_LEN`] bytes
16/// - no `/` (would create a subdirectory in `modules/`)
17/// - no `"` or `\` (would need escaping inside the `.gitmodules`
18///   section name storgit writes for plain-`git` interop)
19/// - no ASCII control characters (`< 0x20` or DEL, including `\0`,
20///   `\n`, `\t`); they would corrupt git tree filenames or the
21///   `.gitmodules` config file
22/// - no leading `.` (rejects `.`, `..`, hidden-file ids)
23/// - does not end in `.git` (collides with the `<id>.git` module dir)
24/// - is not a [reserved name](EntryId::is_reserved) used by storgit
25///   internally (currently just the string `"index"`, which collides
26///   with the parent tree's index subtree)
27///
28/// Holding an [`EntryId`] value is therefore proof that the bytes are
29/// safe to plug into all of storgit's internal paths and tree writes.
30#[derive(Debug, Clone, PartialEq, Eq, PartialOrd, Ord, Hash)]
31pub struct EntryId(String);
32
33/// Reasons [`EntryId::new`] can reject a candidate id.
34#[derive(Debug, Clone, PartialEq, Eq)]
35pub enum EntryIdError {
36    Empty,
37    TooLong { len: usize, max: usize },
38    BadChar(char),
39    LeadingDot,
40    GitSuffix,
41    Reserved,
42}
43
44impl EntryId {
45    /// Maximum byte length of a valid id. Picked to leave headroom
46    /// under typical filesystem name limits (255 bytes on most
47    /// filesystems) once the `<id>.git` suffix is appended.
48    pub const MAX_LEN: usize = 240;
49
50    /// True when `s` is a reserved name storgit uses for its own
51    /// bookkeeping inside the parent tree, and therefore must not be
52    /// used as an entry id.
53    pub fn is_reserved(s: &str) -> bool {
54        s == crate::parent::INDEX_DIR
55    }
56
57    /// Construct a validated id, returning [`EntryIdError`] on rejection.
58    pub fn new(s: impl Into<String>) -> Result<Self, EntryIdError> {
59        let s = s.into();
60        Self::validate(&s)?;
61        Ok(EntryId(s))
62    }
63
64    pub fn as_str(&self) -> &str {
65        &self.0
66    }
67
68    fn validate(s: &str) -> Result<(), EntryIdError> {
69        if s.is_empty() {
70            return Err(EntryIdError::Empty);
71        }
72        if s.len() > Self::MAX_LEN {
73            return Err(EntryIdError::TooLong {
74                len: s.len(),
75                max: Self::MAX_LEN,
76            });
77        }
78        if s.starts_with('.') {
79            return Err(EntryIdError::LeadingDot);
80        }
81        if s.ends_with(".git") {
82            return Err(EntryIdError::GitSuffix);
83        }
84        if Self::is_reserved(s) {
85            return Err(EntryIdError::Reserved);
86        }
87        for c in s.chars() {
88            if c == '/' || c == '"' || c == '\\' {
89                return Err(EntryIdError::BadChar(c));
90            }
91            let code = c as u32;
92            if code < 0x20 || code == 0x7f {
93                return Err(EntryIdError::BadChar(c));
94            }
95        }
96        Ok(())
97    }
98}

106impl std::fmt::Display for EntryIdError {
107    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
108        match self {
109            EntryIdError::Empty => write!(f, "id is empty"),
110            EntryIdError::TooLong { len, max } => write!(f, "id is {len} bytes; max is {max}"),
111            EntryIdError::BadChar(c) => write!(f, "id contains forbidden character {c:?}"),
112            EntryIdError::LeadingDot => write!(f, "id may not start with '.'"),
113            EntryIdError::GitSuffix => write!(f, "id may not end with '.git'"),
114            EntryIdError::Reserved => write!(f, "id is reserved by storgit"),
115        }
116    }
117}

140/// A git commit identifier (SHA-1, 20 bytes).
141#[derive(Debug, Clone, PartialEq, Eq, Hash)]
142pub struct CommitId([u8; 20]);
143
144impl CommitId {
145    /// Number of hex characters used by [`CommitId::to_short_hex`],
146    /// matching git's default abbreviated SHA length.
147    pub const SHORT_HEX_LEN: usize = 7;
148
149    /// Raw SHA-1 bytes.
150    pub fn as_bytes(&self) -> &[u8] {
151        &self.0
152    }
153
154    /// Full 40-character lowercase hex SHA-1, as git prints with `%H`.
155    pub fn to_hex(&self) -> String {
156        let mut s = String::with_capacity(40);
157        for b in &self.0 {
158            s.push_str(&format!("{b:02x}"));
159        }
160        s
161    }
162
163    /// Abbreviated hex SHA-1 (first [`CommitId::SHORT_HEX_LEN`]
164    /// characters), as git prints with `%h`.
165    pub fn to_short_hex(&self) -> String {
166        let mut s = self.to_hex();
167        s.truncate(Self::SHORT_HEX_LEN);
168        s
169    }
170}
171
172impl From<gix::ObjectId> for CommitId {
173    fn from(id: gix::ObjectId) -> Self {
174        let slice = id.as_slice();
175        let mut out = [0u8; 20];
176        out.copy_from_slice(&slice[..20]);
177        CommitId(out)
178    }
